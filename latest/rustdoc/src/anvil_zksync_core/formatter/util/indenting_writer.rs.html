1use std::fmt::Write;
2
3/// IndentingWriter wraps any Write and manages indentation at line starts.
4///
5/// This utility provides an easy way to generate indented text with prefixes
6/// at the beginning of each line. It is useful for formatting output
7/// such as debug information, code, or structured data.
8///
9/// # Examples
10///
11/// ```
12/// use std::fmt::Write;
13/// use anvil_zksync::formatter::util::IndentingWriter;
14///
15/// let mut output = String::new();
16/// let mut writer = IndentingWriter::new(&mut output, Some("> "));
17/// writer.indent();
18/// writeln!(writer, "First line").unwrap();
19/// writeln!(writer, "Second line").unwrap();
20/// writer.dedent();
21/// writeln!(writer, "Third line").unwrap();
22/// ```
23pub struct IndentingWriter<'a, W: Write> {
24    writer: &'a mut W,
25    indent: usize,
26    prefix: Vec<&'a str>,
27    newline: bool,
28}
29
30impl<'a, W: Write> IndentingWriter<'a, W> {
31    const TABULATION: &'static str = "   ";
32
33    /// Creates a new IndentingWriter that wraps the provided writer.
34    ///
35    /// # Arguments
36    ///
37    /// * `writer` - The underlying writer to which formatted text will be written
38    /// * `prefix` - An optional prefix string to add at the beginning of each line after indentation
39    pub fn new(writer: &'a mut W, prefix: Option<&'a str>) -> Self {
40        Self {
41            writer,
42            indent: 0,
43            prefix: prefix.into_iter().collect(),
44            newline: true,
45        }
46    }
47
48    /// Adds a new prefix to the stack of prefixes.
49    ///
50    /// The most recently added prefix will be used for new lines.
51    ///
52    /// # Arguments
53    ///
54    /// * `prefix` - The prefix string to add
55    pub fn push_prefix(&mut self, prefix: &'static str) {
56        self.prefix.push(prefix)
57    }
58
59    /// Removes the most recently added prefix from the stack.
60    ///
61    /// If the stack is empty, this operation does nothing.
62    pub fn pop_prefix(&mut self) {
63        let _ = self.prefix.pop();
64    }
65
66    /// Increases the indentation level by one.
67    pub fn indent(&mut self) {
68        self.indent += 1;
69    }
70
71    /// Decreases the indentation level by one.
72    ///
73    /// If the indentation level is already 0, this operation does nothing.
74    pub fn dedent(&mut self) {
75        if self.indent > 0 {
76            self.indent -= 1;
77        }
78    }
79}
80
81impl<W: Write> IndentingWriter<'_, W> {
82    /// Applies indentation and prefix to the current line if needed.
83    ///
84    /// This is automatically called before writing any content and handles:
85    /// 1. Adding the appropriate number of tabs based on indentation level
86    /// 2. Adding the current prefix (if any)
87    ///
88    /// This method is only applied at the beginning of a line (after a newline).
89    fn do_indent(&mut self) -> std::fmt::Result {
90        if self.newline {
91            for _ in 0..self.indent {
92                self.writer.write_str(Self::TABULATION)?;
93            }
94            self.writer.write_str(self.prefix.last().unwrap_or(&""))?;
95            self.newline = false;
96        }
97
98        Ok(())
99    }
100}
101impl<W: Write> Write for IndentingWriter<'_, W> {
102    fn write_str(&mut self, s: &str) -> std::fmt::Result {
103        let mut start = 0;
104        let mut chars = s.char_indices().peekable();
105
106        while let Some((i, c)) = chars.next() {
107            self.do_indent()?;
108            if c == '\n' {
109                self.newline = true;
110                self.writer.write_str(&s[start..=i])?;
111                start = i + c.len_utf8();
112            } else if chars.peek().is_none() {
113                // If it's the last chunk, write the remainder
114                self.writer.write_str(&s[start..])?;
115            }
116        }
117
118        Ok(())
119    }
120}
