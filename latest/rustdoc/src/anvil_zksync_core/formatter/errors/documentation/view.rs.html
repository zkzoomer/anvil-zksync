19/// Displays the complete documentation for an error.
20///
21/// This view wraps an error and provides a fully-formatted display of all
22/// available documentation, including summary, causes, fixes, references,
23/// and detailed description.
24#[derive(Debug)]
25pub struct FullDocumentation<'a, E>(pub &'a E)
26where
27    E: AnvilErrorDocumentation;
28
29impl<E> PrettyFmt for FullDocumentation<'_, E>
30where
31    E: AnvilErrorDocumentation,
32{
33    fn pretty_fmt(&self, w: &mut impl Write) -> std::fmt::Result {
34        if let Some(doc) = self.0.get_documentation() {
35            format_summary(doc, w)?;
36            if !doc.likely_causes.is_empty() {
37                writeln!(w)?;
38                format_likely_causes(doc, w)?;
39                writeln!(w)?;
40                format_fixes(doc, w)?;
41                writeln!(w)?;
42                format_references(doc, w)?;
43                writeln!(w)?;
44            }
45
46            format_description(doc, w)?;
47        }
48        Ok(())
49    }
50}
51
52impl<E> std::fmt::Display for FullDocumentation<'_, E>
53where
54    E: AnvilErrorDocumentation,
55{
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57        self.pretty_fmt(f)
58    }
59}

61/// Displays only the summary portion of error documentation.
62///
63/// This view provides a concise display showing just the error summary,
64/// which is useful for brief error notifications.
65#[derive(Debug)]
66pub struct SummaryView<'a, E>(pub &'a E)
67where
68    E: AnvilErrorDocumentation;
69
70impl<E> PrettyFmt for SummaryView<'_, E>
71where
72    E: AnvilErrorDocumentation,
73{
74    fn pretty_fmt(&self, w: &mut impl Write) -> std::fmt::Result {
75        if let Some(doc) = self.0.get_documentation() {
76            format_summary(doc, w)?;
77        } else {
78            write!(
79                w,
80                "{error} An unknown error occurred",
81                error = "error:".bright_red()
82            )?;
83        }
84        Ok(())
85    }
86}
87
88impl<E> std::fmt::Display for SummaryView<'_, E>
89where
90    E: AnvilErrorDocumentation,
91{
92    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
93        self.pretty_fmt(f)
94    }
95}
