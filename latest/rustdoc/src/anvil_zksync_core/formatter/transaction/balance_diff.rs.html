31///
32/// Representation of `[BalanceDiff]`, prepared for formatting using `Tabled`
33///
34#[derive(tabled::Tabled)]
35pub(super) struct BalanceDiffRepr {
36    pub address: String,
37    pub before: String,
38    pub after: String,
39    pub delta: String,
40}
41
42fn compute_delta(before: &U256, after: &U256) -> String {
43    match before.cmp(after) {
44        Ordering::Less => format!("+{}", format_gwei(after - before)),
45        Ordering::Equal => "0".to_string(),
46        Ordering::Greater => format!("-{}", format_gwei(before - after)),
47    }
48}
49
50impl From<&BalanceDiff> for BalanceDiffRepr {
51    fn from(val: &BalanceDiff) -> Self {
52        let BalanceDiff {
53            address,
54            balance_before,
55            balance_after,
56        } = val;
57        BalanceDiffRepr {
58            address: address.to_string(),
59            before: format_eth(*balance_before),
60            after: format_eth(*balance_after),
61            delta: compute_delta(balance_before, balance_after),
62        }
63    }
64}
