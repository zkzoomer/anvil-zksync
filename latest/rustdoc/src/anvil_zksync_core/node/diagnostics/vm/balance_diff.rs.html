11///
12/// Holds a part of account state before and after transaction.
13///
14pub struct BalanceDiff {
15    pub address: LabeledAddress,
16    pub balance_before: U256,
17    pub balance_after: U256,
18}
19
20///
21/// Extract information about balances of accounts before and after given
22/// storage logs.
23///
24pub fn extract_balance_diffs(
25    addresses: &HashMap<Address, Option<String>>,
26    log: &Vec<zksync_types::StorageLogWithPreviousValue>,
27) -> Vec<BalanceDiff> {
28    let mut diffs: BTreeMap<StorageKey, internal::BalanceDiffStaging> = addresses
29        .iter()
30        .map(|(a, label)| {
31            (
32                storage_key_for_eth_balance(a),
33                internal::BalanceDiffStaging::new(a, label.clone()),
34            )
35        })
36        .collect();
37
38    for entry in log {
39        if entry.log.is_write() {
40            if let Some(d) = diffs.get_mut(&entry.log.key) {
41                d.balance_before = d.balance_before.or(Some(entry.previous_value));
42                d.balance_after = Some(entry.log.value);
43            }
44        }
45    }
46
47    diffs
48        .into_values()
49        .filter(|d| d.balance_after != d.balance_before)
50        .filter(|d| !address_map::is_system(&d.address.address))
51        .map(Into::into)
52        .collect()
53}

55mod internal {
56    use anvil_zksync_types::traces::LabeledAddress;
57    use zksync_types::{Address, H256, h256_to_u256};
58
59    use super::BalanceDiff;
60
61    ///
62    /// A semi-uninitialized struct to hold a part of account state before and after transaction.
63    ///
64    #[derive(Default)]
65    pub(super) struct BalanceDiffStaging {
66        pub address: LabeledAddress,
67        pub balance_before: Option<H256>,
68        pub balance_after: Option<H256>,
69    }
70
71    impl BalanceDiffStaging {
72        pub fn new(address: &Address, label: Option<String>) -> Self {
73            Self {
74                address: LabeledAddress {
75                    label,
76                    address: *address,
77                },
78                ..Self::default()
79            }
80        }
81    }
82
83    impl From<BalanceDiffStaging> for BalanceDiff {
84        fn from(val: BalanceDiffStaging) -> Self {
85            let BalanceDiffStaging {
86                address,
87                balance_before,
88                balance_after,
89            } = val;
90            BalanceDiff {
91                address,
92                balance_before: h256_to_u256(balance_before.unwrap_or_default()),
93                balance_after: h256_to_u256(balance_after.unwrap_or_default()),
94            }
95        }
96    }
