34pub struct ZKsyncOsVM<S: WriteStorage, H: HistoryMode> {
35    pub storage: StoragePtr<S>,
36    pub tree: InMemoryTree,
37    preimage: InMemoryPreimageSource,
38    transactions: Vec<Transaction>,
39    system_env: SystemEnv,
40    batch_env: L1BatchEnv,
41    config: ZKsyncOsConfig,
42    witness: Option<Vec<u8>>,
43    _phantom: std::marker::PhantomData<H>,
44}

46impl<S: WriteStorage, H: HistoryMode> ZKsyncOsVM<S, H> {
47    pub fn new(
48        batch_env: L1BatchEnv,
49        system_env: SystemEnv,
50        storage: StoragePtr<S>,
51        raw_storage: &InMemoryStorage,
52        config: &ZKsyncOsConfig,
53    ) -> Self {
54        let (tree, preimage) = { create_tree_from_full_state(raw_storage) };
55        ZKsyncOsVM {
56            storage,
57            tree,
58            preimage,
59            transactions: vec![],
60            system_env,
61            batch_env,
62            witness: None,
63            config: config.clone(),
64            _phantom: Default::default(),
65        }
66    }
67
68    /// If any keys are updated in storage externally, but not reflected in internal tree.
69    pub fn update_inconsistent_keys(&mut self, inconsistent_nodes: &[&StorageKey]) {
70        for key in inconsistent_nodes {
71            let value = self.storage.borrow_mut().read_value(key);
72            add_elem_to_tree(&mut self.tree, key, &value);
73        }
74    }
75}

124impl<S: WriteStorage, H: HistoryMode> VmInterface for ZKsyncOsVM<S, H> {
125    type TracerDispatcher = ZkSyncOSTracerDispatcher<S, H>;
126
127    fn push_transaction(
128        &mut self,
129        tx: Transaction,
130    ) -> zksync_multivm::interface::PushTransactionResult<'_> {
131        self.transactions.push(tx);
132        PushTransactionResult {
133            compressed_bytecodes: Default::default(),
134        }
135    }
136
137    fn inspect(
138        &mut self,
139        _dispatcher: &mut Self::TracerDispatcher,
140        execution_mode: zksync_multivm::interface::InspectExecutionMode,
141    ) -> VmExecutionResultAndLogs {
142        if let InspectExecutionMode::Bootloader = execution_mode {
143            // This is called at the end of seal block.
144            // Now is the moment to collect the witness and store it.
145
146            // TODO: add support for multiple transactions.
147
148            if let Some(witness) = self.witness.clone() {
149                set_batch_witness(self.batch_env.number.0, witness);
150            }
151
152            return VmExecutionResultAndLogs {
153                result: ExecutionResult::Success { output: vec![] },
154                logs: Default::default(),
155                statistics: Default::default(),
156                refunds: Default::default(),
157                dynamic_factory_deps: Default::default(),
158            };
159        }
160        let simulate_only = match self.system_env.execution_mode {
161            TxExecutionMode::VerifyExecute => false,
162            TxExecutionMode::EstimateFee => true,
163            TxExecutionMode::EthCall => true,
164        };
165
166        // For now we only support one transaction.
167        assert_eq!(
168            1,
169            self.transactions.len(),
170            "only one tx per batch supported for now"
171        );
172
173        // TODO: add support for multiple transactions.
174        let tx = self.transactions[0].clone();
175        let (result, witness) = execute_tx_in_zkos(
176            &tx,
177            &self.tree,
178            &self.preimage,
179            &mut self.storage,
180            simulate_only,
181            &self.batch_env,
182            self.system_env.chain_id.as_u64(),
183            self.config.zksync_os_bin_path.clone(),
184        );
185
186        self.witness = witness;
187        result
188    }
189
190    fn start_new_l2_block(&mut self, _l2_block_env: zksync_multivm::interface::L2BlockEnv) {
191        // no-op
192    }
