77fn bytes32_to_h256(data: Bytes32) -> H256 {

81fn h256_to_bytes32(data: &H256) -> Bytes32 {

85fn b160_to_h160(data: B160) -> H160 {
86    H160::from_slice(&data.to_be_bytes_vec())

89fn pad_to_word(input: &[u8]) -> Vec<u8> {
90    let mut data = input.to_owned();
91    let remainder = input.len().div_ceil(32) * 32 - input.len();
92    data.extend(std::iter::repeat_n(0u8, remainder));
93    data
94}

96fn h160_to_b160(data: &H160) -> B160 {
97    B160::from_be_bytes(*data.as_fixed_bytes())
98}

100// Helper methods to add data to the Vec<u8> in the format expected by ZKOS.
101fn append_address(data: &mut Vec<u8>, address: &H160) {
102    let mut pp = vec![0u8; 32];
103    let ap1 = address.as_fixed_bytes();
104    pp[12..(20 + 12)].copy_from_slice(ap1);
105    data.append(&mut pp);
106}
107
108fn append_u256(data: &mut Vec<u8>, payload: &zksync_types::U256) {
109    let mut pp = [0u8; 32];
110    payload.to_big_endian(&mut pp);
111
112    data.append(&mut pp.to_vec());
113}
114fn append_u64(data: &mut Vec<u8>, payload: u64) {
115    let mut pp = [0u8; 32];
116    let pp1 = payload.to_be_bytes();
117    pp[24..(8 + 24)].copy_from_slice(&pp1);
118    data.append(&mut pp.to_vec());
119}
120
121fn append_usize(data: &mut Vec<u8>, payload: usize) {
122    let mut pp = [0u8; 32];
123    let pp1 = payload.to_be_bytes();
124    pp[24..(8 + 24)].copy_from_slice(&pp1);
125    data.append(&mut pp.to_vec());
126}

128/// Iterates over raw storage and creates a tree from it.
129pub(super) fn create_tree_from_full_state(
130    raw_storage: &InMemoryStorage,
131) -> (InMemoryTree, InMemoryPreimageSource) {
132    let original_state = &raw_storage.state;
133    let mut tree = InMemoryTree {
134        storage_tree: TestingTree::new_in(Global),
135        cold_storage: HashMap::new(),
136    };
137    let mut preimage_source = InMemoryPreimageSource {
138        inner: HashMap::new(),
139    };
140    for entry in &raw_storage.factory_deps {
141        preimage_source
142            .inner
143            .insert(h256_to_bytes32(entry.0), entry.1.clone());
144    }
145
146    for entry in original_state {
147        let kk = derive_flat_storage_key(
148            &h160_to_b160(entry.0.address()),
149            &h256_to_bytes32(entry.0.key()),
150        );
151        let vv = h256_to_bytes32(entry.1);
152
153        let aa = h256_to_bytes32(entry.0.key());
154        let cc = B160::try_from_be_slice(&aa.as_u8_array()[12..]).unwrap();
155
156        if entry.0.address() == &b160_to_h160(FAKE_NONCE_ADDRESS) {
157            set_account_properties(
158                &mut tree,
159                &mut preimage_source,
160                cc,
161                None,
162                Some(entry.1.to_low_u64_be()),
163                None,
164            );
165        }
166
167        if entry.0.address() == &b160_to_h160(BASE_TOKEN_ADDRESS) {
168            set_account_properties(
169                &mut tree,
170                &mut preimage_source,
171                cc,
172                Some(ruint::aliases::U256::from_be_slice(entry.1.as_bytes())),
173                None,
174                None,
175            );
176        }
177
178        if entry.0.address() == &b160_to_h160(ACCOUNT_CODE_STORAGE_STORAGE_ADDRESS) {
179            println!("Setting bytecode for {:?}", entry.0.key());
180            if entry.0.key().is_zero() {
181                continue;
182            }
183            if entry.1.is_zero() {
184                continue;
185            }
186
187            let bytecode_hash = entry.1;
188            let bytecode = raw_storage
189                .factory_deps
190                .get(bytecode_hash)
191                .unwrap_or_else(|| panic!("Cannot find bytecode for {:?}", bytecode_hash));
192            set_account_properties(
193                &mut tree,
194                &mut preimage_source,
195                cc,
196                None,
197                None,
198                Some(bytecode.clone()),
199            );
200        }
201
202        tree.storage_tree.insert(&kk, &vv);
203        tree.cold_storage.insert(kk, vv);
204    }
205
206    (tree, preimage_source)
207}

217// Serialize Transaction to ZKOS format.
218// Should match the code in basic_bootloader/src/bootloader/transaction/mod.rs
219fn transaction_to_zkos_vec(tx: &Transaction) -> Vec<u8> {
220    let mut tx_raw: Vec<u8> = vec![];
221    let tx_type_id = match tx.tx_format() {
222        zksync_types::l2::TransactionType::LegacyTransaction => 0u8,
223        zksync_types::l2::TransactionType::EIP2930Transaction => 1u8,
224        zksync_types::l2::TransactionType::EIP1559Transaction => 2u8,
225        zksync_types::l2::TransactionType::EIP712Transaction => todo!(),
226        zksync_types::l2::TransactionType::PriorityOpTransaction => todo!(),
227        zksync_types::l2::TransactionType::ProtocolUpgradeTransaction => todo!(),
228    };
229    let common_data = match &tx.common_data {
230        zksync_types::ExecuteTransactionCommon::L1(_) => todo!(),
231        zksync_types::ExecuteTransactionCommon::L2(l2_tx_common_data) => l2_tx_common_data,
232        zksync_types::ExecuteTransactionCommon::ProtocolUpgrade(_) => todo!(),
233    };
234    // tx_type
235    tx_raw.append(&mut vec![0u8; 31]);
236    tx_raw.append(&mut vec![tx_type_id; 1]);
237
238    // from
239    append_address(&mut tx_raw, &common_data.initiator_address);
240    // to
241    append_address(
242        &mut tx_raw,
243        &tx.execute.contract_address.unwrap_or(H160::zero()),
244    );
245
246    let gas_limit = common_data.fee.gas_limit;
247    // gas limit
248    append_u256(&mut tx_raw, &gas_limit);
249    // gas per pubdata limit
250    append_u256(&mut tx_raw, &common_data.fee.gas_per_pubdata_limit);
251
252    let fee_per_gas = common_data.fee.max_fee_per_gas;
253
254    // max fee per gas
255    append_u256(&mut tx_raw, &fee_per_gas);
256    // max priority fee per gas.
257    // TODO: hack for legacy tx (verify!!)
258    append_u256(&mut tx_raw, &common_data.fee.max_priority_fee_per_gas);
259
260    // paymaster
261    append_u64(&mut tx_raw, 0);
262
263    // nonce
264    append_u64(&mut tx_raw, common_data.nonce.0.into());
265
266    append_u256(&mut tx_raw, &tx.execute.value);
267
268    let mut reserved = [0u64; 4];
269
270    // TODO: should check chain_id
271    if tx_type_id == 0 {
272        reserved[0] = 1;
273    }
274
275    if tx.execute.contract_address.is_none() {
276        reserved[1] = 1;
277    }
278
279    for &value in reserved.iter().take(4) {
280        append_u64(&mut tx_raw, value);
281    }
282
283    let signature_u256 = common_data.signature.len().div_ceil(32) as u64;
284
285    let execute_calldata_words = tx.execute.calldata.len().div_ceil(32) as u64;
286
287    let mut current_offset = 19;
288
289    // data offset
290    append_u64(&mut tx_raw, current_offset * 32);
291    // lent
292    current_offset += 1 + execute_calldata_words;
293    // signature offset (stupid -- this doesn't include the padding!!)
294    append_u64(&mut tx_raw, current_offset * 32);
295    current_offset += 1 + signature_u256;
296
297    // factory deps
298    append_u64(&mut tx_raw, current_offset * 32);
299    current_offset += 1;
300    // paymaster
301    append_u64(&mut tx_raw, current_offset * 32);
302    current_offset += 1;
303    // reserved
304    append_u64(&mut tx_raw, current_offset * 32);
305
306    // len - data.
307    append_usize(&mut tx_raw, tx.execute.calldata.len());
308    tx_raw.append(&mut pad_to_word(&tx.execute.calldata));
309
310    // len - signature.
311    append_usize(&mut tx_raw, common_data.signature.len());
312    tx_raw.append(&mut pad_to_word(&common_data.signature));
313
314    // factory deps
315    append_u64(&mut tx_raw, 0);
316    // paymaster
317    append_u64(&mut tx_raw, 0);
318    // reserved
319    append_u64(&mut tx_raw, 0);
320    tx_raw
321}

323#[allow(clippy::too_many_arguments)]
324pub(super) fn execute_tx_in_zkos<W: WriteStorage>(
325    tx: &Transaction,
326    tree: &InMemoryTree,
327    preimage_source: &InMemoryPreimageSource,
328    storage: &mut StoragePtr<W>,
329    simulate_only: bool,
330    batch_env: &L1BatchEnv,
331    chain_id: u64,
332    // if zkos_path is passed, it will also compute witness.
333    zkos_path: Option<String>,
334) -> (VmExecutionResultAndLogs, Option<Vec<u8>>) {
335    let batch_context = zk_ee::system::metadata::BlockMetadataFromOracle {
336        // TODO: get fee from batch_env.
337        eip1559_basefee: ruint::aliases::U256::from(if simulate_only { 0u64 } else { 1000u64 }),
338        block_number: batch_env.number.0 as u64,
339        timestamp: batch_env.timestamp,
340        gas_per_pubdata: ruint::aliases::U256::from(1u64),
341        chain_id,
342        // TODO: set proper values
343        block_hashes: Default::default(),
344        coinbase: B160::ZERO,
345        // TODO: investigate
346        native_price: ruint::aliases::U256::from(10u64),
347        gas_limit: 100_000_000,
348        // TODO: set proper value if needed
349        mix_hash: Default::default(),
350    };
351
352    let storage_commitment = StorageCommitment {
353        root: *tree.storage_tree.root(),
354        next_free_slot: tree.storage_tree.next_free_slot,
355    };
356
357    let mut tx = tx.clone();
358    if simulate_only {
359        // Currently zkos doesn't do validation when running a simulated transaction.
360        // This results in lower gas estimation - which might cause issues for the user.
361        // So here, we're manually adjusting the gas limit to account for the expected validation cost.
362        const ZKOS_EXPECTED_VALIDATION_COST: u64 = 6_000;
363        let new_gas_limit = tx
364            .gas_limit()
365            .saturating_sub(U256::from(ZKOS_EXPECTED_VALIDATION_COST));
366        match &mut tx.common_data {
367            ExecuteTransactionCommon::L1(data) => data.gas_limit = new_gas_limit,
368            ExecuteTransactionCommon::L2(data) => data.fee.gas_limit = new_gas_limit,
369            ExecuteTransactionCommon::ProtocolUpgrade(data) => data.gas_limit = new_gas_limit,
370        };
371
372        // FIXME: currently zkos requires gas price to be >0 (otherwise it fails with out of resources during validation)
373        if tx.max_fee_per_gas() == 0.into() {
374            let new_gas_price = 100.into();
375            if let ExecuteTransactionCommon::L2(data) = &mut tx.common_data {
376                data.fee.max_fee_per_gas = new_gas_price;
377                data.fee.max_priority_fee_per_gas = new_gas_price;
378            };
379        }
380    }
381
382    let tx_raw = transaction_to_zkos_vec(&tx);
383
384    let mut witness = None;
385
386    let (output, dynamic_factory_deps, storage_logs) = if simulate_only {
387        (
388            forward_system::run::simulate_tx(
389                tx_raw,
390                batch_context,
391                tree.clone(),
392                preimage_source.clone(),
393            )
394            .unwrap(),
395            Default::default(), // dynamic factory deps
396            vec![],             // storage logs
397        )
398    } else {
399        let tx_source = TxListSource {
400            transactions: vec![tx_raw].into(),
401        };
402        let noop = NoopTxCallback {};
403        let batch_output = forward_system::run::run_batch(
404            batch_context,
405            // TODO: FIXME
406            tree.clone(),
407            preimage_source.clone(),
408            tx_source.clone(),
409            noop,
410        )
411        .unwrap();
412
413        if let Some(zkos_path) = zkos_path {
414            let result = zksync_os_api::run_batch_generate_witness(
415                batch_context,
416                tree.clone(),
417                preimage_source.clone(),
418                tx_source,
419                storage_commitment,
420                &zkos_path,
421            );
422
423            witness = Some(result.iter().flat_map(|x| x.to_be_bytes()).collect());
424        }
425
426        let mut storage_ptr = storage.borrow_mut();
427
428        let mut storage_logs = vec![];
429
430        // apply storage writes..
431        for write in batch_output.storage_writes {
432            let storage_key = StorageKey::new(
433                AccountTreeId::new(Address::from_slice(&write.account.to_be_bytes_vec())),
434                H256::from(write.account_key.as_u8_array_ref()),
435            );
436            let storage_value = H256::from(write.value.as_u8_array_ref());
437            let prev_value = storage_ptr.set_value(storage_key, storage_value);
438
439            let storage_log = StorageLog {
440                // TODO: FIXME - should distinguish between initial write and repeated write.
441                kind: zksync_types::StorageLogKind::InitialWrite,
442                key: storage_key,
443                value: storage_value,
444            };
445            storage_logs.push(StorageLogWithPreviousValue {
446                log: storage_log,
447                previous_value: prev_value,
448            });
449            if write.account == ACCOUNT_PROPERTIES_STORAGE_ADDRESS {
450                // also update a balance & nonce.
451
452                // find preimage in batch output.published_preimages
453                let dst_address = &write.account_key.as_u8_ref()[12..];
454                let dst_address = zksync_types::Address::from_slice(dst_address);
455                let mut found = false;
456
457                for preimage in batch_output.published_preimages.iter() {
458                    if preimage.0 == write.value {
459                        found = true;
460                        let account_properties =
461                            AccountProperties::decode(&preimage.1.clone().try_into().unwrap());
462                        let balance = account_properties.balance;
463                        let nonce = account_properties.nonce;
464                        let bytecode_hash = account_properties.bytecode_hash;
465
466                        println!(
467                            "For account :{:?} nonce is {:?} balance is {:?}, bytecode_hash is {:?} preimage hash is {:?}",
468                            dst_address, nonce, balance, bytecode_hash, preimage.0
469                        );
470
471                        for (key, value) in [
472                            (
473                                zksync_os_get_nonce_key(&dst_address),
474                                u256_to_h256(U256::from(nonce)),
475                            ),
476                            (
477                                zkos_storage_key_for_standard_token_balance(
478                                    AccountTreeId::new(b160_to_h160(BASE_TOKEN_ADDRESS)),
479                                    &dst_address,
480                                ),
481                                H256::from_slice(&balance.to_be_bytes_vec()),
482                            ),
483                            (
484                                get_code_key(&dst_address),
485                                H256::from(bytecode_hash.as_u8_array_ref()),
486                            ),
487                        ] {
488                            let storage_log = StorageLog {
489                                // TODO: FIXME - should distinguish between initial write and repeated write.
490                                kind: zksync_types::StorageLogKind::InitialWrite,
491                                key,
492                                value,
493                            };
494                            // Create storage write, only if actually something has changed.
495                            let prev_value = storage_ptr.read_value(&key);
496                            if prev_value != value {
497                                let prev_value = storage_ptr.set_value(key, value);
498
499                                storage_logs.push(StorageLogWithPreviousValue {
500                                    log: storage_log,
501                                    previous_value: prev_value,
502                                })
503                            };
504                        }
505                    }
506                }
507                if !found {
508                    panic!("Failed to find preimage for account_properties");
509                }
510            }
511        }
512        let mut f_deps = HashMap::new();
513
514        for factory_dep in batch_output.published_preimages {
515            f_deps.insert(bytes32_to_h256(factory_dep.0), factory_dep.1);
516        }
517
518        (batch_output.tx_results[0].clone(), f_deps, storage_logs)
519    };
520
521    let (tx_output, gas_refunded) = match output.as_ref() {
522        Ok(tx_output) => match &tx_output.execution_result {
523            forward_system::run::ExecutionResult::Success(output) => match &output {
524                forward_system::run::ExecutionOutput::Call(data) => (data, tx_output.gas_refunded),
525                forward_system::run::ExecutionOutput::Create(data, _) => {
526                    (data, tx_output.gas_refunded)
527                }
528            },
529            forward_system::run::ExecutionResult::Revert(data) => {
530                return (
531                    VmExecutionResultAndLogs {
532                        result: ExecutionResult::Revert {
533                            output: VmRevertReason::General {
534                                msg: "Transaction reverted".to_string(),
535                                data: data.clone(),
536                            },
537                        },
538                        logs: VmExecutionLogs {
539                            // TODO: check if we should return storage_logs on revert.
540                            storage_logs,
541                            events: Default::default(),
542                            user_l2_to_l1_logs: Default::default(),
543                            system_l2_to_l1_logs: Default::default(),
544                            total_log_queries_count: Default::default(),
545                        },
546                        statistics: Default::default(),
547                        refunds: Refunds {
548                            gas_refunded: tx_output.gas_refunded,
549                            operator_suggested_refund: tx_output.gas_refunded,
550                        },
551                        dynamic_factory_deps,
552                    },
553                    witness,
554                );
555            }
556        },
557        Err(invalid_tx) => {
558            return (
559                VmExecutionResultAndLogs {
560                    result: ExecutionResult::Revert {
561                        output: VmRevertReason::General {
562                            msg: format!("{:?}", invalid_tx),
563                            data: vec![],
564                        },
565                    },
566                    logs: Default::default(),
567                    statistics: Default::default(),
568                    refunds: Default::default(),
569                    dynamic_factory_deps: Default::default(),
570                },
571                witness,
572            );
573        }
574    };
575
576    (
577        VmExecutionResultAndLogs {
578            result: ExecutionResult::Success {
579                output: tx_output.clone(),
580            },
581            logs: VmExecutionLogs {
582                storage_logs,
583                events: Default::default(),
584                user_l2_to_l1_logs: Default::default(),
585                system_l2_to_l1_logs: Default::default(),
586                total_log_queries_count: Default::default(),
587            },
588            statistics: Default::default(),
589            refunds: Refunds {
590                gas_refunded,
591                operator_suggested_refund: gas_refunded,
592            },
593            dynamic_factory_deps,
594        },
595        witness,
596    )
597}

603/// Create a `key` part of `StorageKey` to access the balance from ERC20 contract balances
604fn zksync_os_key_for_erc20_balance(address: &Address) -> H256 {
605    let address_h256 = address_to_h256(address);
606
607    // 20 bytes address first gets aligned to 32 bytes with index of `balanceOf` storage slot
608    // of default ERC20 contract and to then to 64 bytes.
609    let slot_index = H256::from_low_u64_be(51);
610    let mut bytes = [0_u8; 64];
611    bytes[..32].copy_from_slice(address_h256.as_bytes());
612    bytes[32..].copy_from_slice(slot_index.as_bytes());
613    H256(keccak256(&bytes))
614}

616fn zkos_storage_key_for_standard_token_balance(
617    token_contract: AccountTreeId,
618    address: &Address,
619) -> StorageKey {
620    // We have different implementation of the standard ERC20 contract and native
621    // eth contract. The key for the balance is different for each.
622    let key = if token_contract.address() == &b160_to_h160(BASE_TOKEN_ADDRESS) {
623        zksync_os_key_for_eth_balance(address)
624    } else {
625        zksync_os_key_for_erc20_balance(address)
626    };
627
628    StorageKey::new(token_contract, key)
629}

633fn get_account_properties(
634    state_tree: &InMemoryTree,
635    preimage_source: &mut InMemoryPreimageSource,
636    address: &B160,
637) -> AccountProperties {
638    use forward_system::run::PreimageSource;
639    let key = address_into_special_storage_key(address);
640    let flat_key = derive_flat_storage_key(&ACCOUNT_PROPERTIES_STORAGE_ADDRESS, &key);
641    match state_tree.cold_storage.get(&flat_key) {
642        None => AccountProperties::default(),
643        Some(account_hash) => {
644            if account_hash.is_zero() {
645                // Empty (default) account
646                AccountProperties::default()
647            } else {
648                // Get from preimage:
649                let encoded = preimage_source
650                    .get_preimage(*account_hash)
651                    .unwrap_or_else(|| {
652                        panic!("Missing preimage for {:?} and {:?}", address, account_hash)
653                    });
654
655                AccountProperties::decode(&encoded.try_into().unwrap())
656            }
657        }
658    }
659}
660
661fn set_account_properties(
662    state_tree: &mut InMemoryTree,
663    preimage_source: &mut InMemoryPreimageSource,
664
665    address: B160,
666    balance: Option<ruint::aliases::U256>,
667    nonce: Option<u64>,
668    bytecode: Option<Vec<u8>>,
669) {
670    let mut account_properties = get_account_properties(state_tree, preimage_source, &address);
671    if let Some(bytecode) = bytecode {
672        let bytecode_and_artifacts;
673        (account_properties, bytecode_and_artifacts) =
674            evm_bytecode_into_account_properties(&bytecode);
675        // Save bytecode preimage
676        preimage_source
677            .inner
678            .insert(account_properties.bytecode_hash, bytecode_and_artifacts);
679    }
680    if let Some(nominal_token_balance) = balance {
681        account_properties.balance = nominal_token_balance;
682    }
683    if let Some(nonce) = nonce {
684        account_properties.nonce = nonce;
685    }
686
687    let encoding = account_properties.encoding();
688    let properties_hash = account_properties.compute_hash();
689
690    let key = address_into_special_storage_key(&address);
691    let flat_key = derive_flat_storage_key(&ACCOUNT_PROPERTIES_STORAGE_ADDRESS, &key);
692
693    // Save preimage
694    preimage_source
695        .inner
696        .insert(properties_hash, encoding.to_vec());
697    state_tree.cold_storage.insert(flat_key, properties_hash);
698    state_tree.storage_tree.insert(&flat_key, &properties_hash);
699}

701/// Copied from https://github.com/matter-labs/zksync-os/blob/5e69d4483243bb0d166b7e2137fa54a8bb05925a/tests/rig/src/utils.rs#L426
702/// To avoid bringing in `rig` as a dependency.
703pub fn evm_bytecode_into_account_properties(deployed_code: &[u8]) -> (AccountProperties, Vec<u8>) {
704    use crypto::MiniDigest;
705    use crypto::blake2s::Blake2s256;
706    use crypto::sha3::Keccak256;
707    use evm_interpreter;
708
709    let unpadded_code_len = deployed_code.len();
710    let artifacts =
711        evm_interpreter::BytecodePreprocessingData::create_artifacts_inner(Global, deployed_code);
712    let artifacts = artifacts.as_slice();
713    let artifacts_len = artifacts.len();
714    let padding_len = bytecode_padding_len(unpadded_code_len);
715    let full_len = unpadded_code_len + padding_len + artifacts_len;
716    let mut bytecode_and_artifacts: Vec<u8> = vec![0u8; full_len];
717    bytecode_and_artifacts[..unpadded_code_len].copy_from_slice(deployed_code);
718    let bitmap_offset = unpadded_code_len + padding_len;
719    bytecode_and_artifacts[bitmap_offset..].copy_from_slice(artifacts);
720
721    let observable_bytecode_hash = Bytes32::from_array(Keccak256::digest(deployed_code));
722    let bytecode_hash = Bytes32::from_array(Blake2s256::digest(&bytecode_and_artifacts));
723    let mut result = AccountProperties::TRIVIAL_VALUE;
724    result.observable_bytecode_hash = observable_bytecode_hash;
725    result.bytecode_hash = bytecode_hash;
726    result.versioning_data.set_as_deployed();
727    result
728        .versioning_data
729        .set_ee_version(ExecutionEnvironmentType::EVM as u8);
730    result
731        .versioning_data
732        .set_code_version(evm_interpreter::ARTIFACTS_CACHING_CODE_VERSION_BYTE);
733    result.unpadded_code_len = unpadded_code_len as u32;
734    result.artifacts_len = artifacts_len as u32;
735    result.observable_bytecode_len = unpadded_code_len as u32;
736
737    (result, bytecode_and_artifacts)
738}
