3use zksync_multivm::interface::storage::{ReadStorage, StorageWithOverrides};
4use zksync_types::{
5    AccountTreeId, H256, StorageKey,
6    api::state_override::{BytecodeOverride, OverrideState, StateOverride},
7    bytecode::{BytecodeHash, BytecodeMarker, pad_evm_bytecode},
8    get_code_key, get_evm_code_hash_key, get_known_code_key, get_nonce_key, h256_to_u256,
9    u256_to_h256,
10    utils::{decompose_full_nonce, nonces_to_full_nonce, storage_key_for_eth_balance},
11    web3,
12};
13
14pub(super) fn apply_state_override<S: ReadStorage>(
15    storage: S,
16    state_override: StateOverride,
17) -> StorageWithOverrides<S> {
18    let mut storage = StorageWithOverrides::new(storage);
19
20    for (account, overrides) in state_override {
21        if let Some(balance) = overrides.balance {
22            let balance_key = storage_key_for_eth_balance(&account);
23            storage.set_value(balance_key, u256_to_h256(balance));
24        }
25
26        if let Some(nonce) = overrides.nonce {
27            let nonce_key = get_nonce_key(&account);
28            let full_nonce = storage.read_value(&nonce_key);
29            let (_, deployment_nonce) = decompose_full_nonce(h256_to_u256(full_nonce));
30            let new_full_nonce = u256_to_h256(nonces_to_full_nonce(nonce, deployment_nonce));
31            storage.set_value(nonce_key, new_full_nonce);
32        }
33
34        if let Some(code) = overrides.code {
35            let (bytecode_kind, code) = match code {
36                BytecodeOverride::Unspecified(code) => (BytecodeMarker::detect(&code.0), code),
37                BytecodeOverride::EraVm(code) => (BytecodeMarker::EraVm, code),
38                BytecodeOverride::Evm(code) => (BytecodeMarker::Evm, code),
39            };
40            let code_key = get_code_key(&account);
41
42            let (code_hash, prepared_code) = match bytecode_kind {
43                BytecodeMarker::EraVm => (BytecodeHash::for_bytecode(&code.0).value(), code.0),
44                BytecodeMarker::Evm => {
45                    // For better usability, we allow overriding EVM bytecodes even if EVM contract deployment is not enabled for the chain.
46                    let versioned_hash = BytecodeHash::for_raw_evm_bytecode(&code.0).value();
47                    let evm_bytecode_hash_key = get_evm_code_hash_key(versioned_hash);
48                    storage.set_value(evm_bytecode_hash_key, H256(web3::keccak256(&code.0)));
49                    (versioned_hash, pad_evm_bytecode(&code.0))
50                }
51            };
52
53            storage.set_value(code_key, code_hash);
54            let known_code_key = get_known_code_key(&code_hash);
55            storage.set_value(known_code_key, H256::from_low_u64_be(1));
56            storage.store_factory_dep(code_hash, prepared_code);
57        }
58
59        match overrides.state {
60            Some(OverrideState::State(state)) => {
61                let account = AccountTreeId::new(account);
62                for (key, value) in state {
63                    storage.set_value(StorageKey::new(account, key), value);
64                }
65                storage.insert_erased_account(account);
66            }
67            Some(OverrideState::StateDiff(state_diff)) => {
68                let account = AccountTreeId::new(account);
69                for (key, value) in state_diff {
70                    storage.set_value(StorageKey::new(account, key), value);
71                }
72            }
73            None => { /* do nothing */ }
74        }
75    }
76    storage
77}
