30    pub async fn get_raw_block_transactions_impl(
31        &self,
32        block_number: L2BlockNumber,
33    ) -> Result<Vec<Transaction>, Web3Error> {
34        let tx_hashes = self
35            .blockchain
36            .get_block_tx_hashes_by_number(block_number)
37            .await;
38        if let Some(tx_hashes) = tx_hashes {
39            let mut transactions = Vec::with_capacity(tx_hashes.len());
40            for tx_hash in tx_hashes {
41                let transaction = self
42                    .blockchain
43                    .get_zksync_tx(&tx_hash)
44                    .await
45                    .with_context(|| anyhow::anyhow!("Unexpectedly transaction (hash={tx_hash}) belongs to a block but could not be found"))?;
46                transactions.push(transaction);
47            }
48            Ok(transactions)
49        } else {
50            Ok(self.fork.get_raw_block_transactions(block_number).await?)
51        }
52    }

70    pub async fn get_block_details_impl(
71        &self,
72        block_number: L2BlockNumber,
73    ) -> anyhow::Result<Option<api::BlockDetails>> {
74        let base_system_contracts_hashes = self.system_contracts.base_system_contracts_hashes();
75        let reader = self.inner.read().await;
76        let l2_fair_gas_price = reader.fee_input_provider.fair_l2_gas_price();
77        let fair_pubdata_price = Some(reader.fee_input_provider.fair_pubdata_price());
78        drop(reader);
79
80        let block_details = self
81            .blockchain
82            .get_block_details_by_number(
83                block_number,
84                l2_fair_gas_price,
85                fair_pubdata_price,
86                base_system_contracts_hashes,
87            )
88            .await;
89
90        match block_details {
91            Some(block_details) => Ok(Some(block_details)),
92            None => self.fork.get_block_details(block_number).await,
93        }
94    }

118    pub async fn get_l2_to_l1_log_proof_impl(
119        &self,
120        tx_hash: H256,
121        index: Option<usize>,
122    ) -> anyhow::Result<Option<api::L2ToL1LogProof>> {
123        let Some(tx_receipt) = self.blockchain.get_tx_receipt(&tx_hash).await else {
124            return Ok(None);
125        };
126        let l1_batch_number = L1BatchNumber(tx_receipt.l1_batch_number.expect("").as_u32());
127        let Some(l1_batch) = self.blockchain.get_batch_header(l1_batch_number).await else {
128            return Ok(None);
129        };
130        let all_l1_logs_in_batch = l1_batch
131            .l2_to_l1_logs
132            .into_iter()
133            .map(|log| log.0)
134            .collect::<Vec<_>>();
135        let l1_batch_tx_index = tx_receipt.l1_batch_tx_index.expect("").as_u32() as u16;
136        let log_filter = |log: &L2ToL1Log| log.tx_number_in_block == l1_batch_tx_index;
137        let index_in_filtered_logs = index.unwrap_or(0);
138
139        // Copied from zksync-era
140        let Some((l1_log_index, _)) = all_l1_logs_in_batch
141            .iter()
142            .enumerate()
143            .filter(|(_, log)| log_filter(log))
144            .nth(index_in_filtered_logs)
145        else {
146            return Ok(None);
147        };
148        let merkle_tree_leaves = all_l1_logs_in_batch.iter().map(L2ToL1Log::to_bytes);
149        let tree_size = l2_to_l1_logs_tree_size(self.blockchain.protocol_version());
150
151        let (local_root, proof) = MiniMerkleTree::new(merkle_tree_leaves, Some(tree_size))
152            .merkle_root_and_path(l1_log_index);
153        let Some(aggregated_root) = self
154            .blockchain
155            .get_batch_aggregation_root(l1_batch_number)
156            .await
157        else {
158            return Ok(None);
159        };
160        let root = KeccakHasher.compress(&local_root, &aggregated_root);
161
162        let mut log_leaf_proof = proof;
163        log_leaf_proof.push(aggregated_root);
164
165        let (batch_proof_len, batch_chain_proof, is_final_node) = (0, Vec::<H256>::new(), true);
166
167        let proof = {
168            let mut metadata = [0u8; 32];
169            metadata[0] = LOG_PROOF_SUPPORTED_METADATA_VERSION;
170            metadata[1] = log_leaf_proof.len() as u8;
171            metadata[2] = batch_proof_len as u8;
172            metadata[3] = if is_final_node { 1 } else { 0 };
173
174            let mut result = vec![H256(metadata)];
175
176            result.extend(log_leaf_proof);
177            result.extend(batch_chain_proof);
178
179            result
180        };
181
182        Ok(Some(api::L2ToL1LogProof {
183            proof,
184            root,
185            id: l1_log_index as u32,
186        }))
187    }

189    pub async fn gas_per_pubdata_impl(&self) -> AnvilNodeResult<U256> {
190        let (_, gas_per_pubdata) = self
191            .inner
192            .read()
193            .await
194            .fee_input_provider
195            .gas_price_and_gas_per_pubdata();
196        // We don't accept transactions with `gas_per_pubdata=0` so API should always return 1 at the
197        // bare minimum.
198        let gas_per_pubdata = gas_per_pubdata.max(1);
199        Ok(U256::from(gas_per_pubdata))
200    }
