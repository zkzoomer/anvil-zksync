24impl NodeExecutor {
25    pub fn new(
26        node_inner: Arc<RwLock<InMemoryNodeInner>>,
27        vm_runner: VmRunner,
28        storage_key_layout: StorageKeyLayout,
29    ) -> (Self, NodeExecutorHandle) {
30        let (command_sender, command_receiver) = mpsc::channel(128);
31        let this = Self {
32            node_inner,
33            vm_runner,
34            command_receiver,
35            storage_key_layout,
36        };
37        let handle = NodeExecutorHandle { command_sender };
38        (this, handle)
39    }

41    pub async fn run(mut self) -> AnvilNodeResult<()> {
42        while let Some(command) = self.command_receiver.recv().await {
43            match command {
44                Command::SealBlock(tx_batch, reply) => {
45                    self.seal_block(tx_batch, reply).await?;
46                }
47                Command::SealBlocks(tx_batches, interval, reply) => {
48                    self.seal_blocks(tx_batches, interval, reply).await;
49                }
50                Command::SetCode(address, code, reply) => {
51                    self.set_code(address, code, reply).await;
52                }
53                Command::SetStorage(key, value, reply) => {
54                    self.set_storage(key, value, reply).await;
55                }
56                Command::SetBalance(address, balance, reply) => {
57                    self.set_balance(address, balance, reply).await;
58                }
59                Command::SetNonce(address, nonce, reply) => {
60                    self.set_nonce(address, nonce, reply).await;
61                }
62                Command::ResetFork(url, block_number, reply) => {
63                    self.reset_fork(url, block_number, reply).await;
64                }
65                Command::ResetForkBlockNumber(block_number, reply) => {
66                    self.reset_fork_block_number(block_number, reply).await;
67                }
68                Command::SetForkUrl(url, reply) => {
69                    self.set_fork_url(url, reply).await;
70                }
71                Command::RemoveFork(reply) => {
72                    self.remove_fork(reply).await;
73                }
74                Command::IncreaseTime(delta, reply) => {
75                    self.increase_time(delta, reply).await;
76                }
77                Command::EnforceNextTimestamp(timestamp, reply) => {
78                    self.enforce_next_timestamp(timestamp, reply).await;
79                }
80                Command::SetCurrentTimestamp(timestamp, reply) => {
81                    self.set_current_timestamp(timestamp, reply).await;
82                }
83                Command::SetTimestampInterval(seconds) => {
84                    self.set_timestamp_interval(seconds).await;
85                }
86                Command::RemoveTimestampInterval(reply) => {
87                    self.remove_timestamp_interval(reply).await;
88                }
89                Command::EnforceNextBaseFeePerGas(base_fee, reply) => {
90                    self.enforce_next_base_fee_per_gas(base_fee, reply).await;
91                }
92                Command::SetProgressReport(bar) => {
93                    self.vm_runner.set_progress_report(bar);
94                }
95            }
96        }
97
98        tracing::trace!("channel has been closed; stopping node executor");
99        Ok(())
100    }
101}

103impl NodeExecutor {
104    async fn seal_block(
105        &mut self,
106        tx_batch: TxBatch,
107        reply_sender: Option<oneshot::Sender<AnvilNodeResult<L2BlockNumber>>>,
108    ) -> AnvilNodeResult<()> {
109        let mut node_inner = self.node_inner.write().await;
110        let tx_batch_execution_result = self
111            .vm_runner
112            .run_tx_batch(tx_batch, &mut node_inner)
113            .await?;
114
115        let result = node_inner.seal_block(tx_batch_execution_result).await;
116        drop(node_inner);
117        // Reply to sender if we can, otherwise hold result for further processing
118        let result = if let Some(reply_sender) = reply_sender {
119            if let Err(error_result) = reply_sender.send(result) {
120                tracing::info!("failed to reply as receiver has been dropped");
121                error_result
122            } else {
123                return Ok(());
124            }
125        } else {
126            result
127        };
128        // Not much we can do with an error at this level so we just print it
129        if let Err(err) = result {
130            tracing::error!("failed to seal a block: {:#?}", err);
131        }
132        Ok(())
133    }

135    async fn seal_blocks(
136        &mut self,
137        tx_batches: Vec<TxBatch>,
138        interval: u64,
139        reply: oneshot::Sender<AnvilNodeResult<Vec<L2BlockNumber>>>,
140    ) {
141        let mut node_inner = self.node_inner.write().await;
142
143        // Save old interval to restore later: it might get replaced with `interval` below
144        let old_interval = node_inner.time.get_block_timestamp_interval();
145        let result = async {
146            let mut block_numbers = Vec::with_capacity(tx_batches.len());
147            // Processing the entire vector is essentially atomic here because `NodeExecutor` is
148            // the only component that seals blocks.
149            for (i, tx_batch) in tx_batches.into_iter().enumerate() {
150                // Enforce provided interval starting from the second block (i.e. first block should
151                // use the existing interval).
152                if i == 1 {
153                    node_inner.time.set_block_timestamp_interval(Some(interval));
154                }
155                let tx_batch_execution_result = self
156                    .vm_runner
157                    .run_tx_batch(tx_batch, &mut node_inner)
158                    .await?;
159                let number = node_inner.seal_block(tx_batch_execution_result).await?;
160                block_numbers.push(number);
161            }
162            Ok(block_numbers)
163        }
164        .await;
165        // Restore old interval
166        node_inner.time.set_block_timestamp_interval(old_interval);
167
168        // Reply to sender if we can, otherwise hold result for further processing
169        let result = if let Err(result) = reply.send(result) {
170            tracing::info!("failed to reply as receiver has been dropped");
171            result
172        } else {
173            return;
174        };
175        // Not much we can do with an error at this level so we just print it
176        if let Err(err) = result {
177            tracing::error!("failed to seal blocks: {:#?}", err);
178        }
179    }

181    async fn set_code(
182        &mut self,
183        address: Address,
184        mut bytecode: Vec<u8>,
185        reply: oneshot::Sender<()>,
186    ) {
187        let code_key = get_code_key(&address);
188        let marker = BytecodeMarker::detect(&bytecode);
189        let bytecode_hash = match marker {
190            BytecodeMarker::EraVm => BytecodeHash::for_bytecode(&bytecode),
191            BytecodeMarker::Evm => BytecodeHash::for_raw_evm_bytecode(&bytecode),
192        }
193        .value();
194        if marker == BytecodeMarker::Evm {
195            bytecode = pad_evm_bytecode(&bytecode);
196        }
197
198        // TODO: Likely fork_storage can be moved to `NodeExecutor` instead
199        let node_inner = self.node_inner.read().await;
200        node_inner
201            .fork_storage
202            .store_factory_dep(bytecode_hash, bytecode);
203        node_inner.fork_storage.set_value(code_key, bytecode_hash);
204        drop(node_inner);
205        // Reply to sender if we can
206        if reply.send(()).is_err() {
207            tracing::info!("failed to reply as receiver has been dropped");
208        }
209    }

600async fn execute_without_response(
601    command_sender: &mpsc::Sender<Command>,
602    command: Command,
603) -> AnvilNodeResult<()> {
604    let action_name = command.readable_action_name();
605    command_sender
606        .send(command)
607        .await
608        .map_err(|_| node_executor_dropped_error("request to", &action_name))
609}
610
611async fn execute_with_response<R>(
612    command_sender: &mpsc::Sender<Command>,
613    command_gen: impl FnOnce(oneshot::Sender<R>) -> Command,
614) -> AnvilNodeResult<R> {
615    let (response_sender, response_receiver) = oneshot::channel();
616
617    let command = command_gen(response_sender);
618    let action_name = command.readable_action_name();
619    command_sender
620        .send(command)
621        .await
622        .map_err(|_| node_executor_dropped_error("request to", &action_name))?;
623
624    match response_receiver.await {
625        Ok(result) => Ok(result),
626        Err(_) => Err(node_executor_dropped_error(
627            "receive a response to the request to",
628            &action_name,
629        )),
630    }
631}
632fn node_executor_dropped_error(request_or_receive: &str, action_name: &str) -> AnvilNodeError {
633    anvil_zksync::node::generic_error!(
634        r"Failed to {request_or_receive} {action_name} because node executor is dropped. \
635         Another error was likely propagated from the main execution loop. \
636         If this is not the case, please, report this as a bug."
637    )
638}

640#[derive(Debug)]
641enum Command {
642    // Block sealing commands
643    SealBlock(
644        TxBatch,
645        Option<oneshot::Sender<AnvilNodeResult<L2BlockNumber>>>,
646    ),
647    SealBlocks(
648        Vec<TxBatch>,
649        u64,
650        oneshot::Sender<AnvilNodeResult<Vec<L2BlockNumber>>>,
651    ),
652    // Storage manipulation commands
653    SetCode(Address, Vec<u8>, oneshot::Sender<()>),
654    SetStorage(StorageKey, U256, oneshot::Sender<()>),
655    SetBalance(Address, U256, oneshot::Sender<()>),
656    SetNonce(Address, U256, oneshot::Sender<()>),
657    // Fork manipulation commands
658    ResetFork(
659        Url,
660        Option<L2BlockNumber>,
661        oneshot::Sender<AnvilNodeResult<()>>,
662    ),
663    ResetForkBlockNumber(L2BlockNumber, oneshot::Sender<AnvilNodeResult<()>>),
664    SetForkUrl(Url, oneshot::Sender<Option<Url>>),
665    RemoveFork(oneshot::Sender<()>),
666    // Time manipulation commands. Caveat: reply-able commands can hold user connections alive for
667    // a long time (until the command is processed).
668    IncreaseTime(u64, oneshot::Sender<()>),
669    EnforceNextTimestamp(u64, oneshot::Sender<AnvilNodeResult<()>>),
670    SetCurrentTimestamp(u64, oneshot::Sender<i128>),
671    SetTimestampInterval(u64),
672    RemoveTimestampInterval(oneshot::Sender<bool>),
673    // Fee manipulation commands
674    EnforceNextBaseFeePerGas(U256, oneshot::Sender<()>),
675    // Replay tx progress indicator
676    SetProgressReport(Option<ProgressBar>),
677}
678
679impl Command {
680    /// Human-readable command description used for diagnostics.
681    fn readable_action_name(&self) -> String {
682        fn batch_repr(batch: &TxBatch) -> String {
683            format!(
684                "{:?}",
685                batch
686                    .txs
687                    .iter()
688                    .map(|tx| tx.hash().to_string())
689                    .collect::<Vec<_>>()
690            )
691        }
692        match self {
693            Command::SealBlock(tx_batch, _) => {
694                format!("seal a block with transactions {}", batch_repr(tx_batch))
695            }
696            Command::SealBlocks(vec, interval, _) => format!(
697                "seal blocks with intervals of {interval} seconds between consecutive blocks: {:?}",
698                vec.iter().map(batch_repr).collect::<Vec<_>>()
699            ),
700            Command::SetCode(address, _bytecode, _) => {
701                format!("set bytecode for address {address}")
702            }
703            Command::SetStorage(storage_key, value, _) => {
704                format!("set storage {}={value}", storage_key.hashed_key())
705            }
706            Command::SetBalance(account, new_value, _) => {
707                format!("set balance of account {account} to {new_value}")
708            }
709            Command::SetNonce(account, nonce, _) => {
710                format!("set nonce of account {account} to {nonce}")
711            }
712            Command::ResetFork(url, l2_block_number, _) => {
713                format!("reset fork to url {url} and block number {l2_block_number:?}")
714            }
715            Command::ResetForkBlockNumber(l2_block_number, _) => {
716                format!("reset fork block number to {l2_block_number}")
717            }
718            Command::SetForkUrl(url, _) => format!("set fork RPC URL to {url}"),
719            Command::RemoveFork(_) => "remove fork if there was one".into(),
720            Command::IncreaseTime(delta, _) => format!("increase time by {delta} seconds"),
721            Command::EnforceNextTimestamp(timestamp, _) => {
722                format!("enforce next block's timestamp to {timestamp} seconds")
723            }
724            Command::SetCurrentTimestamp(timestamp, _) => {
725                format!("set current timestamp to {timestamp} seconds")
726            }
727            Command::SetTimestampInterval(interval) => {
728                format!("set block timestamp interval to {interval} seconds")
729            }
730            Command::RemoveTimestampInterval(_) => "remove timestamp interval".into(),
731            Command::EnforceNextBaseFeePerGas(base_fee, _) => {
732                format!("enforce next block's base fee per gas to {base_fee}")
733            }
734            Command::SetProgressReport(_) => "set progress report".into(),
735        }
736    }
737}

739#[cfg(test)]
740pub mod testing {
741    use super::*;
742    use backon::{ConstantBuilder, ExponentialBuilder, Retryable};
743    use std::time::Duration;
744    use tokio::sync::mpsc::error::TryRecvError;
745
746    pub struct NodeExecutorTester {
747        receiver: Arc<RwLock<mpsc::Receiver<Command>>>,
748    }
749
750    impl NodeExecutorTester {
751        pub fn new() -> (Self, NodeExecutorHandle) {
752            let (command_sender, command_receiver) = mpsc::channel(128);
753            (
754                Self {
755                    receiver: Arc::new(RwLock::new(command_receiver)),
756                },
757                NodeExecutorHandle { command_sender },
758            )
759        }
760
761        async fn recv(&self) -> anyhow::Result<Command> {
762            let mut receiver = self.receiver.write().await;
763            tokio::time::timeout(Duration::from_millis(100), receiver.recv())
764                .await
765                .map_err(|_| anyhow::anyhow!("no command received"))
766                .and_then(|res| res.ok_or(anyhow::anyhow!("disconnected")))
767        }
768
769        /// Assert that the next command is sealing provided tx batch. Waits with a timeout for the
770        /// next command to arrive if the command queue is empty.
771        pub async fn expect_seal_block(&self, expected_tx_batch: TxBatch) -> anyhow::Result<()> {
772            let command = (|| self.recv())
773                .retry(ExponentialBuilder::default())
774                .await?;
775            match command {
776                Command::SealBlock(actual_tx_batch, _) if actual_tx_batch == expected_tx_batch => {
777                    Ok(())
778                }
779                _ => anyhow::bail!("unexpected command: {:?}", command),
780            }
781        }
782
783        /// Assert that the next command is sealing provided tx batch. Unlike `expect_seal_block`
784        /// this method does not retry.
785        pub async fn expect_seal_block_immediate(
786            &self,
787            expected_tx_batch: TxBatch,
788        ) -> anyhow::Result<()> {
789            let result = self.receiver.write().await.try_recv();
790            match result {
791                Ok(Command::SealBlock(actual_tx_batch, _))
792                    if actual_tx_batch == expected_tx_batch =>
793                {
794                    Ok(())
795                }
796                Ok(command) => anyhow::bail!("unexpected command: {:?}", command),
797                Err(TryRecvError::Empty) => anyhow::bail!("no command received"),
798                Err(TryRecvError::Disconnected) => anyhow::bail!("disconnected"),
799            }
800        }
801
802        /// Assert that there are no command currently in receiver queue. Waits up to a timeout for
803        /// the next command to potentially arrive.
804        pub async fn expect_empty(&self) -> anyhow::Result<()> {
805            let result = (|| self.recv())
806                .retry(
807                    ConstantBuilder::default()
808                        .with_delay(Duration::from_millis(100))
809                        .with_max_times(3),
810                )
811                .await;
812            match result {
813                Ok(command) => {
814                    anyhow::bail!("unexpected command: {:?}", command)
815                }
816                Err(err) if err.to_string().contains("no command received") => Ok(()),
817                Err(err) => anyhow::bail!("unexpected error: {:?}", err),
818            }
819        }
820
821        /// Assert that there are no command currently in receiver queue. Unlike `expect_empty`
822        /// this method does not retry.
823        pub async fn expect_empty_immediate(&self) -> anyhow::Result<()> {
824            let result = self.receiver.write().await.try_recv();
825            match result {
826                Ok(command) => {
827                    anyhow::bail!("unexpected command: {:?}", command)
828                }
829                Err(TryRecvError::Empty) => Ok(()),
830                Err(TryRecvError::Disconnected) => anyhow::bail!("disconnected"),
831            }
832        }
833    }
834}
