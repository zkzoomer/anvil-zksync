29/// Trait that provides necessary data when forking a remote chain.
30///
31/// Most methods' signatures are similar to corresponding methods from [`EthNamespaceClient`] and [`ZksNamespaceClient`]
32/// but with domain-specific types where that makes sense. Additionally, return types are wrapped
33/// into [`anyhow::Result`] to avoid leaking RPC-specific implementation details.
34#[async_trait]
35pub trait ForkSource: fmt::Debug + Send + Sync {
36    /// Alternative for [`Clone::clone`] that is object safe.
37    fn dyn_cloned(&self) -> Box<dyn ForkSource>;
38
39    /// Human-readable description on the fork's origin. None if there is no fork.
40    fn url(&self) -> Option<Url>;
41
42    /// Details on the fork's state at the moment when we forked from it. None if there is no fork.
43    fn details(&self) -> Option<ForkDetails>;
44
45    /// Fetches fork's storage value at a given index for given address for the forked blocked.
46    async fn get_storage_at(
47        &self,
48        address: Address,
49        idx: U256,
50        block: Option<api::BlockIdVariant>,
51    ) -> anyhow::Result<H256>;
52
53    /// Fetches fork's storage value at a given index for given address for the forked blocked.
54    async fn get_storage_at_forked(&self, address: Address, idx: U256) -> anyhow::Result<H256>;
55
56    /// Returns the bytecode stored under this hash (if available).
57    async fn get_bytecode_by_hash(&self, hash: H256) -> anyhow::Result<Option<Vec<u8>>>;
58
59    /// Fetches fork's transaction for a given hash.
60    async fn get_transaction_by_hash(&self, hash: H256)
61    -> anyhow::Result<Option<api::Transaction>>;
62
63    /// Fetches fork's transaction details for a given hash.
64    async fn get_transaction_details(
65        &self,
66        hash: H256,
67    ) -> anyhow::Result<Option<api::TransactionDetails>>;
68
69    /// Fetches fork's transactions that belong to a block with the given number.
70    async fn get_raw_block_transactions(
71        &self,
72        block_number: L2BlockNumber,
73    ) -> anyhow::Result<Vec<zksync_types::Transaction>>;
74
75    /// Fetches fork's block for a given hash.
76    async fn get_block_by_hash(
77        &self,
78        hash: H256,
79    ) -> anyhow::Result<Option<api::Block<api::TransactionVariant>>>;
80
81    /// Fetches fork's block for a given number.
82    async fn get_block_by_number(
83        &self,
84        block_number: api::BlockNumber,
85    ) -> anyhow::Result<Option<api::Block<api::TransactionVariant>>>;
86
87    /// Fetches fork's block for a given id.
88    async fn get_block_by_id(
89        &self,
90        block_id: api::BlockId,
91    ) -> anyhow::Result<Option<api::Block<api::TransactionVariant>>> {
92        match block_id {
93            api::BlockId::Hash(hash) => self.get_block_by_hash(hash).await,
94            api::BlockId::Number(number) => self.get_block_by_number(number).await,
95        }
96    }
97
98    /// Fetches fork's block details for a given block number.
99    async fn get_block_details(
100        &self,
101        block_number: L2BlockNumber,
102    ) -> anyhow::Result<Option<api::BlockDetails>>;
103
104    /// Fetches fork's transaction count for a given block hash.
105    async fn get_block_transaction_count_by_hash(
106        &self,
107        block_hash: H256,
108    ) -> anyhow::Result<Option<U256>>;
109
110    /// Fetches fork's transaction count for a given block number.
111    async fn get_block_transaction_count_by_number(
112        &self,
113        block_number: api::BlockNumber,
114    ) -> anyhow::Result<Option<U256>>;
115
116    /// Fetches fork's transaction count for a given block id.
117    async fn get_block_transaction_count_by_id(
118        &self,
119        block_id: api::BlockId,
120    ) -> anyhow::Result<Option<U256>> {
121        match block_id {
122            api::BlockId::Hash(hash) => self.get_block_transaction_count_by_hash(hash).await,
123            api::BlockId::Number(number) => {
124                self.get_block_transaction_count_by_number(number).await
125            }
126        }
127    }
128
129    /// Fetches fork's transaction by block hash and transaction index position.
130    async fn get_transaction_by_block_hash_and_index(
131        &self,
132        block_hash: H256,
133        index: Index,
134    ) -> anyhow::Result<Option<api::Transaction>>;
135
136    /// Fetches fork's transaction by block number and transaction index position.
137    async fn get_transaction_by_block_number_and_index(
138        &self,
139        block_number: api::BlockNumber,
140        index: Index,
141    ) -> anyhow::Result<Option<api::Transaction>>;
142
143    /// Fetches fork's transaction by block id and transaction index position.
144    async fn get_transaction_by_block_id_and_index(
145        &self,
146        block_id: api::BlockId,
147        index: Index,
148    ) -> anyhow::Result<Option<api::Transaction>> {
149        match block_id {
150            api::BlockId::Hash(hash) => {
151                self.get_transaction_by_block_hash_and_index(hash, index)
152                    .await
153            }
154            api::BlockId::Number(number) => {
155                self.get_transaction_by_block_number_and_index(number, index)
156                    .await
157            }
158        }
159    }
160
161    /// Fetches fork's addresses of the default bridge contracts.
162    async fn get_bridge_contracts(&self) -> anyhow::Result<Option<api::BridgeAddresses>>;
163}

230impl ForkClient {
231    async fn new(
232        config: ForkConfig,
233        l2_client: Box<DynClient<L2>>,
234        block_number: L2BlockNumber,
235    ) -> anyhow::Result<Self> {
236        let ForkConfig {
237            url,
238            estimate_gas_price_scale_factor,
239            estimate_gas_scale_factor,
240        } = config;
241        let chain_id = l2_client
242            .chain_id()
243            .await
244            .with_context(|| format!("failed to get chain id from fork={url}"))?;
245        let chain_id = L2ChainId::try_from(chain_id.as_u64())
246            .map_err(|e| anyhow::anyhow!("fork has malformed chain id: {e}"))?;
247        let block_details = l2_client
248            .get_block_details(block_number)
249            .await?
250            .ok_or_else(
251                || anyhow::anyhow!("could not find block #{block_number} at fork={url}",),
252            )?;
253        let root_hash = block_details
254            .base
255            .root_hash
256            .ok_or_else(|| anyhow::anyhow!("fork block #{block_number} missing root hash"))?;
257        let mut block = l2_client
258            .get_block_by_hash(root_hash, true)
259            .await?
260            .ok_or_else(|| {
261                anyhow::anyhow!(
262                    "could not find API block #{block_number} at fork={url} despite finding its details \
263                    through `zks_getBlockDetails` previously; this is likely a bug, please report this",
264                )
265            })?;
266        let batch_number = block_details.l1_batch_number;
267        // TODO: This is a bit weird, we should just grab last block from the latest sealed L1 batch
268        //       instead to ensure `l1BatchNumber` is always present.
269        block.l1_batch_number = Some(batch_number.0.into());
270
271        let Some(protocol_version) = block_details.protocol_version else {
272            // It is possible that some external nodes do not store protocol versions for versions below 9.
273            // That's why we assume that whenever a protocol version is not present, it is unsupported by anvil-zksync.
274            anyhow::bail!(
275                "Block #{block_number} from fork={url} does not have protocol version set. \
276                Likely you are using an external node with a block for protocol version below 9 which are unsupported in anvil-zksync. \
277                Please report this as a bug if that's not the case."
278            )
279        };
280        if !SupportedProtocolVersions::is_supported(protocol_version) {
281            anyhow::bail!(
282                "Block #{block_number} from fork={url} is using unsupported protocol version `{protocol_version}`. \
283                    anvil-zksync only supports the following versions: {SupportedProtocolVersions}."
284            )
285        }
286
287        let fee_params = l2_client.get_fee_params().await?;
288        let details = ForkDetails {
289            chain_id,
290            protocol_version,
291            batch_number,
292            block_number,
293            block_hash: root_hash,
294            block_timestamp: block_details.base.timestamp,
295            api_block: block,
296            l1_gas_price: block_details.base.l1_gas_price,
297            l2_fair_gas_price: block_details.base.l2_fair_gas_price,
298            fair_pubdata_price: block_details
299                .base
300                .fair_pubdata_price
301                .unwrap_or(DEFAULT_FAIR_PUBDATA_PRICE),
302            estimate_gas_price_scale_factor,
303            estimate_gas_scale_factor,
304            fee_params,
305        };
306        let fork = ForkClient {
307            url,
308            details,
309            l2_client,
310        };
311        Ok(fork)
312    }

314    /// Initializes a fork based on config at a given block number.
315    pub async fn at_block_number(
316        config: ForkConfig,
317        block_number: Option<L2BlockNumber>,
318    ) -> anyhow::Result<Self> {
319        let l2_client =
320            zksync_web3_decl::client::Client::http(SensitiveUrl::from(config.url.clone()))?.build();
321        let block_number = if let Some(block_number) = block_number {
322            block_number
323        } else {
324            let block_number = l2_client
325                .get_block_number()
326                .await
327                .with_context(|| format!("failed to get block number from fork={}", config.url))?;
328            L2BlockNumber(block_number.as_u32())
329        };
330
331        Self::new(config, Box::new(l2_client), block_number).await
332    }
333
334    /// Initializes a fork based on config at a block BEFORE given transaction.
335    /// This will allow us to apply this transaction locally on top of this fork.
336    pub async fn at_before_tx(
337        config: ForkConfig,
338        tx_hash: H256,
339    ) -> anyhow::Result<(Self, Vec<Transaction>)> {
340        let l2_client =
341            zksync_web3_decl::client::Client::http(SensitiveUrl::from(config.url.clone()))?.build();
342        let tx_details = l2_client
343            .get_transaction_by_hash(tx_hash)
344            .await?
345            .ok_or_else(|| {
346                anyhow::anyhow!(
347                    "could not find tx with hash={tx_hash:?} at fork={}",
348                    config.url
349                )
350            })?;
351        let block_number = tx_details.block_number.ok_or_else(|| {
352            anyhow::anyhow!(
353                "could not initialize fork from tx with hash={tx_hash:?} as it is still pending"
354            )
355        })?;
356        let block_number = L2BlockNumber(block_number.as_u32());
357        if block_number == L2BlockNumber(0) {
358            anyhow::bail!(
359                "could not initialize fork from tx with hash={tx_hash:?} as it belongs to genesis"
360            );
361        }
362        let mut earlier_txs = Vec::new();
363        for tx in l2_client.get_raw_block_transactions(block_number).await? {
364            let hash = tx.hash();
365            earlier_txs.push(tx);
366
367            if hash == tx_hash {
368                break;
369            }
370        }
371
372        // We initialize fork from the parent of the block containing transaction.
373        Ok((
374            Self::new(config, Box::new(l2_client), block_number - 1).await?,
375            earlier_txs,
376        ))
377    }
378}

465#[derive(Debug)]
466struct ForkState {
467    client: Option<ForkClient>,
468    cache: Cache,
469}
470
471impl Fork {
472    pub(super) fn new(client: Option<ForkClient>, cache_config: CacheConfig) -> Self {
473        let cache = Cache::new(cache_config);
474        Self {
475            state: Arc::new(RwLock::new(ForkState { client, cache })),
476        }
477    }
478
479    pub(super) fn reset_fork_client(&self, client: Option<ForkClient>) {
480        // TODO: We don't clean cache here so it might interfere with the new fork. Consider
481        //       parametrizing cache by fork URL to avoid this.
482        self.write().client = client;
483    }
484
485    pub(super) fn set_fork_url(&self, new_url: Url) -> Option<Url> {
486        // We are assuming that the new url is pointing to the same logical data source so we do not
487        // invalidate cache
488        let mut writer = self.write();
489        if let Some(client) = writer.client.as_mut() {
490            Some(std::mem::replace(&mut client.url, new_url))
491        } else {
492            None
493        }
494    }
495
496    fn read(&self) -> RwLockReadGuard<ForkState> {
497        self.state.read().expect("Fork lock is poisoned")
498    }
499
500    fn write(&self) -> RwLockWriteGuard<ForkState> {
501        self.state.write().expect("Fork lock is poisoned")
502    }
503
504    async fn make_call<T, F: Future<Output = anyhow::Result<T>>>(
505        &self,
506        method: &str,
507        call_body: impl FnOnce(Box<DynClient<L2>>) -> F,
508    ) -> Option<anyhow::Result<T>> {
509        let (client, span) = if let Some(client) = self.read().client.as_ref() {
510            let span = tracing::info_span!("fork_rpc_call", method, url = %client.url);
511            (client.l2_client.clone(), span)
512        } else {
513            return None;
514        };
515        Some(
516            call_body(client)
517                .map_err(|error| {
518                    sh_err!("call failed: {}", error);
519                    error
520                })
521                .instrument(span)
522                .await,
523        )
524    }
525}

544    async fn get_storage_at(
545        &self,
546        address: Address,
547        idx: U256,
548        block: Option<api::BlockIdVariant>,
549    ) -> anyhow::Result<H256> {
550        // TODO: This is currently cached at the `ForkStorage` level but I am unsure if this is a
551        //       good thing. Intuitively it feels like cache should be centralized in a single place.
552        self.make_call("get_storage_at", |client| async move {
553            client
554                .get_storage_at(address, idx, block)
555                .await
556                .with_context(|| format!("(address={address:?}, idx={idx:?})"))
557        })
558        .await
559        .unwrap_or(Ok(H256::zero()))
560    }
561
562    async fn get_storage_at_forked(&self, address: Address, idx: U256) -> anyhow::Result<H256> {
563        let Some(block_number) = self
564            .read()
565            .client
566            .as_ref()
567            .map(|client| client.details.block_number)
568        else {
569            return Ok(H256::zero());
570        };
571        self.get_storage_at(
572            address,
573            idx,
574            Some(api::BlockIdVariant::BlockNumber(api::BlockNumber::Number(
575                block_number.0.into(),
576            ))),
577        )
578        .await
579    }

581    async fn get_bytecode_by_hash(&self, hash: H256) -> anyhow::Result<Option<Vec<u8>>> {
582        // TODO: This is currently cached at the `ForkStorage` level but I am unsure if this is a
583        //       good thing. Intuitively it feels like cache should be centralized in a single place.
584        self.make_call("get_bytecode_by_hash", |client| async move {
585            client
586                .get_bytecode_by_hash(hash)
587                .await
588                .with_context(|| format!("(hash={hash:?})"))
589        })
590        .await
591        .unwrap_or(Ok(None))
592    }
593
594    async fn get_transaction_by_hash(
595        &self,
596        hash: H256,
597    ) -> anyhow::Result<Option<api::Transaction>> {
598        if let Some(tx) = self.read().cache.get_transaction(&hash).cloned() {
599            tracing::debug!(?hash, "using cached transaction");
600            return Ok(Some(tx));
601        }
602
603        let tx = self
604            .make_call("get_transaction_by_hash", |client| async move {
605                client
606                    .get_transaction_by_hash(hash)
607                    .await
608                    .with_context(|| format!("(hash={hash:?})"))
609            })
610            .await
611            .unwrap_or(Ok(None))?;
612
613        if let Some(tx) = tx {
614            self.write().cache.insert_transaction(hash, tx.clone());
615            Ok(Some(tx))
616        } else {
617            Ok(None)
618        }
619    }
620
621    async fn get_transaction_details(
622        &self,
623        hash: H256,
624    ) -> anyhow::Result<Option<api::TransactionDetails>> {
625        // N.B. We don't cache these responses as they will change through the lifecycle of the transaction
626        // and caching could be error-prone. In theory, we could cache responses once the txn status
627        // is `final` or `failed` but currently this does not warrant the additional complexity.
628        self.make_call("get_transaction_details", |client| async move {
629            client
630                .get_transaction_details(hash)
631                .await
632                .with_context(|| format!("(hash={hash:?})"))
633        })
634        .await
635        .unwrap_or(Ok(None))
636    }

638    async fn get_raw_block_transactions(
639        &self,
640        block_number: L2BlockNumber,
641    ) -> anyhow::Result<Vec<zksync_types::Transaction>> {
642        if let Some(txs) = self
643            .read()
644            .cache
645            .get_block_raw_transactions(&(block_number.0 as u64))
646            .cloned()
647        {
648            tracing::debug!(%block_number, "using cached block raw transactions");
649            return Ok(txs);
650        }
651
652        let txs = self
653            .make_call("get_raw_block_transactions", |client| async move {
654                client
655                    .get_raw_block_transactions(block_number)
656                    .await
657                    .with_context(|| format!("(block_number={block_number})"))
658            })
659            .await
660            .unwrap_or(Err(Web3Error::NoBlock.into()))?;
661
662        self.write()
663            .cache
664            .insert_block_raw_transactions(block_number.0 as u64, txs.clone());
665        Ok(txs)
666    }

668    async fn get_block_by_hash(
669        &self,
670        hash: H256,
671    ) -> anyhow::Result<Option<api::Block<api::TransactionVariant>>> {
672        if let Some(block) = self.read().cache.get_block(&hash, true).cloned() {
673            tracing::debug!(?hash, "using cached block");
674            return Ok(Some(block));
675        }
676
677        let block = self
678            .make_call("get_block_by_hash", |client| async move {
679                client
680                    .get_block_by_hash(hash, true)
681                    .await
682                    .with_context(|| format!("(hash={hash:?}, full_transactions=true)"))
683            })
684            .await
685            .unwrap_or(Ok(None))?;
686
687        if let Some(block) = block {
688            self.write().cache.insert_block(hash, true, block.clone());
689            Ok(Some(block))
690        } else {
691            Ok(None)
692        }
693    }
694
695    async fn get_block_by_number(
696        &self,
697        block_number: api::BlockNumber,
698    ) -> anyhow::Result<Option<api::Block<api::TransactionVariant>>> {
699        match block_number {
700            api::BlockNumber::Number(block_number) => {
701                {
702                    let guard = self.read();
703                    let cache = &guard.cache;
704                    if let Some(block) = cache
705                        .get_block_hash(&block_number.as_u64())
706                        .and_then(|hash| cache.get_block(hash, true))
707                        .cloned()
708                    {
709                        tracing::debug!(%block_number, "using cached block");
710                        return Ok(Some(block));
711                    }
712                }
713
714                let block = self
715                    .make_call("get_block_by_number", |client| async move {
716                        client
717                            .get_block_by_number(api::BlockNumber::Number(block_number), true)
718                            .await
719                            .with_context(|| {
720                                format!("(block_number={block_number}, full_transactions=true)")
721                            })
722                    })
723                    .await
724                    .unwrap_or(Ok(None))?;
725
726                if let Some(block) = block {
727                    self.write()
728                        .cache
729                        .insert_block(block.hash, true, block.clone());
730                    Ok(Some(block))
731                } else {
732                    Ok(None)
733                }
734            }
735            _ => self
736                .make_call("get_block_by_number", |client| async move {
737                    client
738                        .get_block_by_number(block_number, true)
739                        .await
740                        .with_context(|| {
741                            format!("(block_number={block_number}, full_transactions=true)")
742                        })
743                })
744                .await
745                .unwrap_or(Ok(None)),
746        }
747    }

837    async fn get_bridge_contracts(&self) -> anyhow::Result<Option<api::BridgeAddresses>> {
838        if let Some(bridge_contracts) = self.read().cache.get_bridge_addresses().cloned() {
839            tracing::debug!("using cached bridge contracts");
840            return Ok(Some(bridge_contracts));
841        }
842
843        let bridge_contracts = self
844            .make_call("get_bridge_contracts", |client| async move {
845                Ok(Some(client.get_bridge_contracts().await?))
846            })
847            .await
848            .unwrap_or(Ok(None))?;
849
850        if let Some(bridge_contracts) = bridge_contracts {
851            self.write()
852                .cache
853                .set_bridge_addresses(bridge_contracts.clone());
854            Ok(Some(bridge_contracts))
855        } else {
856            Ok(None)
857        }
858    }

861struct SupportedProtocolVersions;
862
863impl SupportedProtocolVersions {
864    const SUPPORTED_VERSIONS: [ProtocolVersionId; 3] = [
865        ProtocolVersionId::Version26,
866        ProtocolVersionId::Version27,
867        ProtocolVersionId::Version28,
868    ];
869
870    fn is_supported(version: ProtocolVersionId) -> bool {
871        Self::SUPPORTED_VERSIONS.contains(&version)
872    }
873}
874
875impl fmt::Display for SupportedProtocolVersions {
876    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
877        f.write_str(
878            &Self::SUPPORTED_VERSIONS
879                .iter()
880                .map(|v| v.to_string())
881                .join(", "),
882        )
883    }
