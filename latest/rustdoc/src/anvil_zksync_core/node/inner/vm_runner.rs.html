111        bootloader_debug_result: Option<&eyre::Result<BootloaderDebug, String>>,
112        spent_on_pubdata: u64,

123            Ok(())
124        } else {
125            Err("Bootloader tracer didn't finish.".to_owned())
126        }

129    /// Validates L2 transaction
130    fn validate_tx(
131        &self,
132        batch_env: &L1BatchEnv,
133        tx_data: &L2TxCommonData,
134    ) -> Result<(), anvil_zksync::tx_invalid::TransactionValidationError> {
135        let max_gas = U256::from(u64::MAX);
136        if tx_data.fee.gas_limit > max_gas {
137            return Err(anvil_zksync::tx_invalid::InvalidGasLimit {
138                tx_gas_limit: Box::new(tx_data.fee.gas_limit),
139                max_gas: Box::new(max_gas),
140            });
141        }
142
143        if tx_data.fee.gas_per_pubdata_limit > max_gas {
144            return Err(anvil_zksync::tx_invalid::GasPerPubdataLimit {
145                tx_gas_per_pubdata_limit: Box::new(tx_data.fee.gas_per_pubdata_limit),
146                max_gas: Box::new(max_gas),
147            });
148        }
149
150        let l2_gas_price = batch_env.fee_input.fair_l2_gas_price();
151        if tx_data.fee.max_fee_per_gas < l2_gas_price.into() {
152            return Err(anvil_zksync::tx_invalid::MaxFeePerGasTooLow {
153                max_fee_per_gas: Box::new(tx_data.fee.max_fee_per_gas),
154                l2_gas_price: Box::new(l2_gas_price.into()),
155            });
156        }
157
158        if tx_data.fee.max_fee_per_gas < tx_data.fee.max_priority_fee_per_gas {
159            return Err(anvil_zksync::tx_invalid::MaxPriorityFeeGreaterThanMaxFee {
160                max_fee_per_gas: Box::new(tx_data.fee.max_fee_per_gas),
161                max_priority_fee_per_gas: Box::new(tx_data.fee.max_priority_fee_per_gas),
162            });
163        }
164        Ok(())
165    }

167    async fn run_tx_pretty(
168        &mut self,
169        tx: &Transaction,
170        executor: &mut dyn BatchExecutor<ForkStorage>,
171        config: &TestNodeConfig,
172        fee_model_config: &FeeModelConfigV2,
173    ) -> AnvilNodeResult<BatchTransactionExecutionResult> {
174        let verbosity = get_shell().verbosity;
175
176        let BatchTransactionExecutionResult {
177            tx_result,
178            compression_result,
179            call_traces,
180        } = executor.execute_tx(tx.clone()).await?;
181        compression_result.map_err(|_inner| {
182            // We ignore `inner` because bytecode
183            // compression error currently does not hold
184            // any precise information
185            anvil_zksync::node::TransactionHalt {
186                inner: Box::new(anvil_zksync::halt::FailedToPublishCompressedBytecodes),
187                transaction_hash: Box::new(tx.hash()),
188            }
189        })?;
190
191        let spent_on_pubdata =
192            tx_result.statistics.gas_used - tx_result.statistics.computational_gas_used as u64;
193
194        let mut known_addresses = known_addresses_after_transaction(tx);
195        let mut trace_output = None;
196
197        if !call_traces.is_empty() {
198            let mut builder = CallTraceDecoderBuilder::default();
199
200            builder = builder.with_signature_identifier(SignaturesIdentifier::global());
201
202            let decoder = builder.build();
203            let mut arena = build_call_trace_arena(&call_traces, &tx_result);
204            extract_addresses(&arena, &mut known_addresses);
205
206            if verbosity >= 2 {
207                decode_trace_arena(&mut arena, &decoder).await;
208                let filtered_arena = filter_call_trace_arena(&arena, verbosity);
209                trace_output = Some(render_trace_arena_inner(&filtered_arena, false));
210            }
211        }
212
213        let balance_diffs: Vec<formatter::transaction::balance_diff::BalanceDiff> =
214            extract_balance_diffs(&known_addresses, &tx_result.logs.storage_logs)
215                .into_iter()
216                .map(Into::into)
217                .collect();
218
219        sh_println!(
220            "{}",
221            TransactionSummary::new(
222                config.get_l2_gas_price(),
223                tx,
224                &tx_result,
225                (verbosity >= 1).then_some(balance_diffs),
226            )
227        );
228
229        if let Some(trace_output) = trace_output {
230            sh_println!("\nTraces:\n{}", trace_output);
231        }
232
233        // Print gas details if enabled
234        if config.show_gas_details != ShowGasDetails::None {
235            self.display_detailed_gas_info(
236                Some(&self.bootloader_debug_result.read().unwrap()),
237                spent_on_pubdata,
238                fee_model_config,
239            )
240            .unwrap_or_else(|err| {
241                sh_err!("{}", format!("Cannot display gas details: {err}"));
242            });
243        }
244        // Print storage logs if enabled
245        if config.show_storage_logs != ShowStorageLogs::None {
246            print_storage_logs_details(config.show_storage_logs, &tx_result);
247        }
248        // Print VM details if enabled
249        if config.show_vm_details != ShowVMDetails::None {
250            let mut formatter = Formatter::new();
251            formatter.print_vm_details(&tx_result);
252        }
253
254        Ok(BatchTransactionExecutionResult {
255            tx_result,
256            compression_result: Ok(()),
257            call_traces,
258        })
259    }

261    /// Runs transaction and commits it to a new block.
262    #[allow(clippy::too_many_arguments)]
263    async fn run_tx(
264        &mut self,
265        tx: &Transaction,
266        tx_index: u64,
267        next_log_index: &mut usize,
268        block_ctx: &BlockContext,
269        batch_env: &L1BatchEnv,
270        executor: &mut dyn BatchExecutor<ForkStorage>,
271        config: &TestNodeConfig,
272        fee_model_config: &FeeModelConfigV2,
273        impersonating: bool,
274    ) -> AnvilNodeResult<TransactionResult> {
275        let tx_hash = tx.hash();
276        let transaction_type = tx.tx_format();
277
278        if let ExecuteTransactionCommon::L2(l2_tx_data) = &tx.common_data {
279            // If the transaction can not be validated, we return immediately
280            self.validate_tx(batch_env, l2_tx_data).map_err(|e| {
281                anvil_zksync::node::TransactionValidationFailed {
282                    inner: Box::new(e),
283                    transaction_hash: Box::new(tx_hash),
284                }
285            })?;
286        }
287
288        let BatchTransactionExecutionResult {
289            tx_result: result,
290            compression_result: _,
291            call_traces,
292        } = self
293            .run_tx_pretty(tx, executor, config, fee_model_config)
294            .await?;
295
296        if let ExecutionResult::Halt { reason } = result.result {
297            // Halt means that something went really bad with the transaction execution
298            // (in most cases invalid signature, but it could also be bootloader panic etc).
299            // In such cases, we should not persist the VM data and should pretend that
300            // the transaction never existed.
301            return Err(anvil_zksync::node::TransactionHalt {
302                inner: Box::new(reason.to_halt_error().await),
303                transaction_hash: Box::new(tx_hash),
304            });
305        }
306
307        if impersonating {
308            // During impersonation, we skip account validation (which is responsible for updating nonce)
309            // so we do it manually for each transaction that didn't result in a halt.
310            let nonce_key = self.storage_layout.get_nonce_key(&tx.initiator_account());
311            let nonce = h256_to_u256(self.fork_storage.read_value(&nonce_key));
312            let nonce = u256_to_h256(nonce + 1);
313            self.fork_storage.set_value(nonce_key, nonce);
314        }
315
316        let mut new_bytecodes = new_bytecodes(tx, &result);
317
318        if self.system_contracts.zksync_os.zksync_os {
319            // In zksync_os, we store account properties outside of state (so state has only hash).
320            // For now, we simply put the original preimages into the factory deps.
321            // The result type here is the 'era' crate - that is not modified to fit zksync_os yet.
322            // once it is - we will not need this hack anymore.
323            new_bytecodes.extend(result.dynamic_factory_deps.clone());
324        }
325
326        let logs = result
327            .logs
328            .events
329            .iter()
330            .enumerate()
331            .map(|(log_idx, log)| api::Log {
332                address: log.address,
333                topics: log.indexed_topics.clone(),
334                data: Bytes(log.value.clone()),
335                block_hash: Some(block_ctx.hash),
336                block_number: Some(block_ctx.miniblock.into()),
337                l1_batch_number: Some(U64::from(batch_env.number.0)),
338                transaction_hash: Some(tx_hash),
339                transaction_index: Some(U64::from(tx_index)),
340                log_index: Some(U256::from(log_idx)),
341                transaction_log_index: Some(U256::from(log_idx)),
342                log_type: None,
343                removed: Some(false),
344                block_timestamp: Some(block_ctx.timestamp.into()),
345            })
346            .collect();
347        let base_fee = get_batch_base_fee(batch_env, VmVersion::latest());
348        let effective_gas_price = match &tx.common_data {
349            ExecuteTransactionCommon::L1(l1_common_data) => Some(l1_common_data.max_fee_per_gas),
350            ExecuteTransactionCommon::L2(l2_common_data) => {
351                Some(l2_common_data.fee.get_effective_gas_price(base_fee.into()))
352            }
353            ExecuteTransactionCommon::ProtocolUpgrade(upgrade_common_data) => {
354                Some(upgrade_common_data.max_fee_per_gas)
355            }
356        };
357
358        let tx_receipt = api::TransactionReceipt {
359            transaction_hash: tx_hash,
360            transaction_index: U64::from(tx_index),
361            block_hash: block_ctx.hash,
362            block_number: block_ctx.miniblock.into(),
363            l1_batch_tx_index: Some(U64::from(tx_index)),
364            l1_batch_number: Some(U64::from(batch_env.number.0)),
365            from: tx.initiator_account(),
366            to: tx.recipient_account(),
367            cumulative_gas_used: Default::default(),
368            gas_used: Some(tx.gas_limit() - result.refunds.gas_refunded),
369            contract_address: contract_address_from_tx_result(&result),
370            logs,
371            l2_to_l1_logs: result
372                .logs
373                .user_l2_to_l1_logs
374                .iter()
375                .enumerate()
376                .map(|(log_index, log)| api::L2ToL1Log {
377                    block_hash: Some(block_ctx.hash),
378                    block_number: block_ctx.miniblock.into(),
379                    l1_batch_number: Some(U64::from(batch_env.number.0)),
380                    log_index: U256::from(*next_log_index + log_index),
381                    transaction_index: U64::from(tx_index),
382                    transaction_hash: tx_hash,
383                    transaction_log_index: U256::from(log_index),
384                    tx_index_in_l1_batch: Some(U64::from(tx_index)),
385                    shard_id: log.0.shard_id.into(),
386                    is_service: log.0.is_service,
387                    sender: log.0.sender,
388                    key: log.0.key,
389                    value: log.0.value,
390                })
391                .collect(),
392            status: if result.result.is_failed() {
393                U64::from(0)
394            } else {
395                U64::from(1)
396            },
397            effective_gas_price,
398            transaction_type: Some((transaction_type as u32).into()),
399            logs_bloom: Default::default(),
400        };
401        *next_log_index += result.logs.user_l2_to_l1_logs.len();
402        let debug = create_debug_output(tx, &result, call_traces).expect("create debug output"); // OK to unwrap here as Halt is handled above
403
404        Ok(TransactionResult {
405            info: TxExecutionInfo {
406                tx: tx.clone(),
407                batch_number: batch_env.number.0,
408                miniblock_number: block_ctx.miniblock,
409            },
410            new_bytecodes,
411            receipt: tx_receipt,
412            debug,
413        })
414    }

416    pub(super) async fn run_tx_batch(
417        &mut self,
418        TxBatch { txs, impersonating }: TxBatch,
419        node_inner: &mut InMemoryNodeInner,
420    ) -> AnvilNodeResult<TxBatchExecutionResult> {
421        let system_contracts = self
422            .system_contracts
423            .contracts(TxExecutionMode::VerifyExecute, impersonating)
424            .clone();
425        let base_system_contracts_hashes = system_contracts.hashes();
426        // Prepare a new block context and a new batch env
427        let system_env =
428            node_inner.create_system_env(system_contracts, TxExecutionMode::VerifyExecute);
429        let (batch_env, mut block_ctx) = node_inner.create_l1_batch_env().await;
430        // Advance clock as we are consuming next timestamp for this block
431
432        if self.time.advance_timestamp() != block_ctx.timestamp {
433            return Err(anvil_zksync::node::generic_error!(
434                "Advancing clock produced different timestamp than expected. This should never happen -- please report this as a bug."
435            ));
436        };
437
438        let pubdata_params = PubdataParams {
439            l2_da_validator_address: Address::zero(),
440            pubdata_type: PubdataType::Rollup,
441        };
442        let mut executor = if self.system_contracts.zksync_os.zksync_os {
443            self.executor_factory.init_main_batch(
444                self.fork_storage.clone(),
445                batch_env.clone(),
446                system_env.clone(),
447                pubdata_params,
448                // For zksync_os, we have to pass the iterator handle to the storage
449                // as zksync_os has different storage layout, so it has to scan over whole storage.
450                Some(self.fork_storage.inner.read().unwrap().raw_storage.clone()),
451            )
452        } else {
453            self.executor_factory.init_main_batch(
454                self.fork_storage.clone(),
455                batch_env.clone(),
456                system_env.clone(),
457                pubdata_params,
458                None,
459            )
460        };
461
462        // Compute block hash. Note that the computed block hash here will be different than that in production.
463        let tx_hashes = txs.iter().map(|t| t.hash()).collect::<Vec<_>>();
464        block_ctx.hash = compute_hash(
465            system_env.version,
466            (block_ctx.miniblock as u32).into(),
467            block_ctx.timestamp,
468            block_ctx.prev_block_hash,
469            &tx_hashes,
470        );
471
472        // Execute transactions and bootloader
473        let mut tx_results = Vec::with_capacity(tx_hashes.len());
474        let mut tx_index = 0;
475        let mut next_log_index = 0;
476        let total = txs.len();
477
478        for tx in txs {
479            if let Some(ref pb) = self.progress_report {
480                pb.set_message(format!(
481                    "Replaying transaction {}/{} from 0x{:x}...",
482                    tx_index + 1,
483                    total,
484                    tx.hash()
485                ));
486            }
487
488            let result = self
489                .run_tx(
490                    &tx,
491                    tx_index,
492                    &mut next_log_index,
493                    &block_ctx,
494                    &batch_env,
495                    &mut executor,
496                    &node_inner.config,
497                    &node_inner.fee_input_provider.get_fee_model_config(),
498                    impersonating,
499                )
500                .await;
501
502            // Update progress bar
503            if let Some(ref pb) = self.progress_report {
504                pb.inc(1);
505            }
506            match result {
507                Ok(tx_result) => {
508                    tx_results.push(tx_result);
509                    tx_index += 1;
510                }
511                Err(e) => {
512                    match &e {
513                        // Validation errors are reported and the execution proceeds
514                        AnvilNodeError::TransactionValidationFailed { .. } => {
515                            let error_report = ExecutionErrorReport::new(&e, &tx);
516                            sh_eprintln!("{error_report}");
517                            executor.rollback_last_tx().await?;
518                        }
519                        // Halts are reported and the execution proceeds
520                        AnvilNodeError::TransactionHalt { inner, .. } => {
521                            let error_report = ExecutionErrorReport::new(inner.as_ref(), &tx);
522                            sh_eprintln!("{error_report}");
523                            executor.rollback_last_tx().await?;
524                        }
525                        // Other errors are not recoverable so we pass them up
526                        // the execution stack immediately
527                        _ => return Err(e),
528                    }
529                }
530            }
531        }
532        // TODO: This is the correct hash as reported by VM, but we can't compute it correct above
533        //       because we don't know which txs are going to be halted
534        block_ctx.hash = compute_hash(
535            system_env.version,
536            (block_ctx.miniblock as u32).into(),
537            block_ctx.timestamp,
538            block_ctx.prev_block_hash,
539            tx_results
540                .iter()
541                .map(|tx_result| &tx_result.receipt.transaction_hash),
542        );
543
544        let mut block_ctxs = vec![block_ctx.clone()];
545        if !tx_results.is_empty() {
546            // Create an empty virtual block at the end of the batch (only if the last block was
547            // not empty, i.e. virtual).
548            let mut virtual_block_ctx = block_ctx.new_block(&mut self.time);
549            virtual_block_ctx.hash = L2BlockHasher::new(
550                L2BlockNumber(virtual_block_ctx.miniblock as u32),
551                virtual_block_ctx.timestamp,
552                block_ctx.hash,
553            )
554            .finalize(system_env.version);
555            let l2_block_env = L2BlockEnv {
556                number: (block_ctx.miniblock + 1) as u32,
557                timestamp: block_ctx.timestamp + 1,
558                prev_block_hash: block_ctx.hash,
559                max_virtual_blocks_to_create: 1,
560            };
561            executor.start_next_l2_block(l2_block_env).await?;
562            block_ctxs.push(virtual_block_ctx);
563        }
564
565        let (finished_l1_batch, modified_storage_keys) = if self.generate_system_logs {
566            // If system log generation is enabled we run realistic (and time-consuming) bootloader flow
567            let (finished_l1_batch, storage_view) = Box::new(executor).finish_batch().await?;
568            (
569                finished_l1_batch,
570                storage_view.modified_storage_keys().clone(),
571            )
572        } else {
573            // Otherwise we mock the execution with a single bootloader iteration
574            let mut finished_l1_batch = FinishedL1Batch::mock();
575            let (bootloader_execution_result, storage_view) = executor.bootloader().await?;
576            finished_l1_batch.block_tip_execution_result = bootloader_execution_result;
577            (
578                finished_l1_batch,
579                storage_view.modified_storage_keys().clone(),
580            )
581        };
582        assert!(
583            !finished_l1_batch
584                .block_tip_execution_result
585                .result
586                .is_failed(),
587            "VM must not fail when finalizing block: {:#?}",
588            finished_l1_batch.block_tip_execution_result.result
589        );
590
591        Ok(TxBatchExecutionResult {
592            tx_results,
593            base_system_contracts_hashes,
594            batch_env,
595            block_ctxs,
596            finished_l1_batch,
597            modified_storage_keys,
598        })
599    }

607fn new_bytecodes(
608    tx: &Transaction,
609    result: &VmExecutionResultAndLogs,
610) -> Vec<(zksync_types::H256, Vec<u8>)> {
611    let saved_factory_deps = VmEvent::extract_bytecodes_marked_as_known(&result.logs.events);
612
613    // Get transaction factory deps
614    let factory_deps = &tx.execute.factory_deps;
615    let mut tx_factory_deps: HashMap<_, _> = factory_deps
616        .iter()
617        .map(|bytecode| {
618            (
619                BytecodeHash::for_bytecode(bytecode).value(),
620                bytecode.clone(),
621            )
622        })
623        .collect();
624    // Ensure that *dynamic* factory deps (ones that may be created when executing EVM contracts)
625    // are added into the lookup map as well.
626    tx_factory_deps.extend(result.dynamic_factory_deps.clone());
627    saved_factory_deps
628        .map(|bytecode_hash| {
629            let bytecode = tx_factory_deps.get(&bytecode_hash).unwrap_or_else(|| {
630                panic!(
631                    "Failed to get factory deps on tx: bytecode hash: {:?}, tx hash: {}",
632                    bytecode_hash,
633                    tx.hash()
634                )
635            });
636            (bytecode_hash, bytecode.clone())
637        })
638        .collect::<Vec<_>>()
639}

641fn contract_address_from_tx_result(execution_result: &VmExecutionResultAndLogs) -> Option<H160> {
642    for query in execution_result.logs.storage_logs.iter().rev() {
643        if query.log.is_write() && query.log.key.address() == &ACCOUNT_CODE_STORAGE_ADDRESS {
644            return Some(h256_to_address(query.log.key.key()));
645        }
646    }
647    None
648}

817    /// Decodes a `bytes` tx result to its concrete parameter type.
818    fn decode_tx_result(output: &[u8], param_type: DynSolType) -> DynSolValue {
819        let result = DynSolType::Bytes
820            .abi_decode(output)
821            .expect("failed decoding output");
822        let result_bytes = match result {
823            DynSolValue::Bytes(bytes) => bytes,
824            _ => panic!("expected bytes but got a different type"),
825        };
826
827        param_type
828            .abi_decode(&result_bytes)
829            .expect("failed decoding output")
830    }
