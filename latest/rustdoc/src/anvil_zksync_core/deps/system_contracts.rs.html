48static BUILTIN_CONTRACT_ARTIFACTS: Lazy<HashMap<ProtocolVersionId, HashMap<String, Vec<u8>>>> =
49    Lazy::new(|| {
50        let mut result = HashMap::new();
51        for (protocol_version, built_in_contracts) in BUILTIN_CONTRACT_ARCHIVES {
52            let decoder = GzDecoder::new(built_in_contracts);
53            let mut archive = tar::Archive::new(decoder);
54            let mut contract_artifacts = HashMap::new();
55            for file in archive
56                .entries()
57                .expect("failed to decompress built-in contracts")
58            {
59                let mut file = file.expect("failed to read a built-in contract entry");
60                let path = file
61                    .header()
62                    .path()
63                    .expect("contract path is malformed")
64                    .file_name()
65                    .expect("built-in contract entry does not have a filename")
66                    .to_string_lossy()
67                    .to_string();
68
69                let mut contents = Vec::with_capacity(
70                    file.header().size().expect("contract size is corrupted") as usize,
71                );
72                file.read_to_end(&mut contents).unwrap();
73                contract_artifacts.insert(path, contents);
74            }
75            result.insert(protocol_version, contract_artifacts);
76        }
77        result
78    });

80pub fn bytecode_from_slice(artifact_name: &str, contents: &[u8]) -> Vec<u8> {
81    let artifact: Value = serde_json::from_slice(contents).expect(artifact_name);
82    let bytecode = artifact["bytecode"]
83        .as_object()
84        .unwrap_or_else(|| panic!("Bytecode not found in {artifact_name:?}"))
85        .get("object")
86        .unwrap_or_else(|| panic!("Bytecode object not found in {artifact_name:?}"))
87        .as_str()
88        .unwrap_or_else(|| panic!("Bytecode object is not a string in {artifact_name:?}"));
89
90    hex::decode(bytecode)
91        .unwrap_or_else(|err| panic!("Can't decode bytecode in {artifact_name:?}: {err}"))
92}

94pub fn load_builtin_contract(protocol_version: ProtocolVersionId, artifact_name: &str) -> Vec<u8> {
95    let artifact_path = format!("{artifact_name}.json");
96    bytecode_from_slice(
97        artifact_name,
98        BUILTIN_CONTRACT_ARTIFACTS
99            .get(&protocol_version)
100            .unwrap_or_else(|| panic!("protocol version '{protocol_version}' is not supported"))
101            .get(&artifact_path)
102            .unwrap_or_else(|| {
103                panic!("failed to find built-in contract artifact at '{artifact_path}'")
104            }),
105    )
106}

108/// Build a static map of “everything” (kernel + non-kernel + precompile + L2 + empty).
109static BUILTIN_CONTRACTS: Lazy<HashMap<ProtocolVersionId, Vec<DeployedContract>>> =
110    Lazy::new(|| {
111        let mut result = HashMap::new();
112
113        for (protocol_version, _) in BUILTIN_CONTRACT_ARCHIVES {
114            let mut list = Vec::new();
115
116            // (1) Kernel
117            list.extend(
118                BUILTIN_CONTRACT_LOCATIONS
119                    .iter()
120                    .filter(|(_, _, min_version)| &protocol_version >= min_version)
121                    .map(|(artifact_name, address, _)| DeployedContract {
122                        account_id: AccountTreeId::new(*address),
123                        bytecode: load_builtin_contract(protocol_version, artifact_name),
124                    }),
125            );
126
127            list.extend(
128                NON_KERNEL_CONTRACT_LOCATIONS
129                    .iter()
130                    .filter(|(_, _, min_version)| &protocol_version >= min_version)
131                    .map(|(artifact_name, address, _)| DeployedContract {
132                        account_id: AccountTreeId::new(*address),
133                        bytecode: load_builtin_contract(protocol_version, artifact_name),
134                    }),
135            );
136
137            result.insert(protocol_version, list);
138        }
139
140        result
141    });

218pub fn get_deployed_contracts(
219    options: SystemContractsOptions,
220    protocol_version: ProtocolVersionId,
221    system_contracts_path: Option<&Path>,
222) -> Vec<DeployedContract> {
223    match options {
224        SystemContractsOptions::BuiltIn | SystemContractsOptions::BuiltInWithoutSecurity => {
225            BUILTIN_CONTRACTS
226                .get(&protocol_version)
227                .unwrap_or_else(|| panic!("protocol version '{protocol_version}' is not supported"))
228                .clone()
229        }
230        SystemContractsOptions::Local => {
231            // checks if system contracts path is provided
232            if let Some(path) = system_contracts_path {
233                get_system_smart_contracts_from_dir(path.to_path_buf())
234            } else {
235                get_system_smart_contracts()
236            }
237        }
238    }
239}

245    fn count_protocol_contracts(protocol_version: ProtocolVersionId) -> usize {
246        let kernel_count = BUILTIN_CONTRACT_LOCATIONS
247            .iter()
248            .filter(|(_, _, min_version)| &protocol_version >= min_version)
249            .count();
250
251        let non_kernel_count = NON_KERNEL_CONTRACT_LOCATIONS
252            .iter()
253            .filter(|(_, _, min_version)| &protocol_version >= min_version)
254            .count();
255
256        kernel_count + non_kernel_count
257    }
