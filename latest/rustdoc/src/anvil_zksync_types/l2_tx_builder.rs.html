1use zksync_multivm::zk_evm_latest::ethereum_types::U256;
2use zksync_types::api::TransactionRequest;

4use zksync_types::l2::L2Tx;
5use zksync_types::transaction_request::PaymasterParams;
6use zksync_types::{Address, K256PrivateKey, L2ChainId, Nonce, PackedEthSignature};
7
8pub struct L2TxBuilder {
9    from: Address,
10    to: Option<Address>,
11    nonce: Nonce,
12    value: U256,
13    gas_limit: U256,
14    max_fee_per_gas: U256,
15    max_priority_fee_per_gas: U256,
16    calldata: Vec<u8>,
17    factory_deps: Vec<Vec<u8>>,
18    chain_id: L2ChainId,
19}
20
21impl L2TxBuilder {
22    pub fn new(
23        from: Address,
24        nonce: Nonce,
25        gas_limit: U256,
26        max_fee_per_gas: U256,
27        chain_id: L2ChainId,
28    ) -> Self {
29        Self {
30            from,
31            to: None,
32            nonce,
33            value: U256::zero(),
34            gas_limit,
35            max_fee_per_gas,
36            max_priority_fee_per_gas: max_fee_per_gas,
37            calldata: vec![],
38            factory_deps: vec![],
39            chain_id,
40        }
41    }
42
43    pub fn with_to(mut self, to: Address) -> Self {
44        self.to = Some(to);
45        self
46    }
47
48    pub fn with_calldata(mut self, calldata: Vec<u8>) -> Self {
49        self.calldata = calldata;
50        self
51    }
52
53    pub fn with_max_priority_fee_per_gas(mut self, max_priority_fee_per_gas: U256) -> Self {
54        self.max_priority_fee_per_gas = max_priority_fee_per_gas;
55        self
56    }
57
58    pub fn build_impersonated(self) -> L2Tx {
59        let mut tx = L2Tx::new(
60            self.to,
61            self.calldata,
62            self.nonce,
63            Fee {
64                gas_limit: self.gas_limit,
65                max_fee_per_gas: self.max_fee_per_gas,
66                max_priority_fee_per_gas: self.max_priority_fee_per_gas,
67                gas_per_pubdata_limit: U256::from(50_000),
68            },
69            self.from,
70            self.value,
71            self.factory_deps,
72            PaymasterParams::default(),
73        );
74        let mut req: TransactionRequest = tx.clone().into();
75        req.chain_id = Some(self.chain_id.as_u64());
76        let data = req.get_default_signed_message().unwrap();
77        let sig = PackedEthSignature::sign_raw(&K256PrivateKey::random(), &data).unwrap();
78        let raw = req.get_signed_bytes(&sig).unwrap();
79        let (_, hash) = TransactionRequest::from_bytes_unverified(&raw).unwrap();
80
81        tx.set_input(raw, hash);
82        tx.common_data.signature = sig.serialize_packed().into();
83        tx
84    }
85}
